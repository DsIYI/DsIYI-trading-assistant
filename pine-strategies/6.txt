//@version=6
strategy('SuperTrend STRATEGY', overlay = true, pyramiding = 1)

// Timing settings
FromMonth = input.int(defval = 5, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(defval = 24, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(defval = 2024, title = 'From Year', minval = 2024)
ToMonth = input.int(defval = 1, title = 'To Month', minval = 1, maxval = 12)
ToDay = input.int(defval = 1, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(defval = 9999, title = 'To Year', minval = 2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish

// Input parameters
Periods = input(title = 'ATR Period', defval = 10)
src = input(hl2, title = 'Source')
Multiplier = input.float(title = 'ATR Multiplier', step = 0.1, defval = 3.0)
changeATR = input(title = 'Change ATR Calculation Method ?', defval = true)
showsignals = input(title = 'Show Buy/Sell Signals ?', defval = true)
highlighting = input(title = 'Highlighter On/Off ?', defval = true)
barcoloring = input(title = 'Bar Coloring On/Off ?', defval = true)
trailingStop = input.float(title = 'Trailing Stop Loss (%)', defval = 1.5, step = 0.1)
sensitivity = input.float(title = 'Trend Sensitivity', defval = 0.5, step = 0.1, minval = 0.1, maxval = 1.0)

// ATR calculations
atr2 = ta.sma(ta.tr, Periods)
atr = changeATR ? ta.atr(Periods) : atr2
up = src - Multiplier * atr * sensitivity
up1 = nz(up[1], up)
up := close[1] > up1 ? math.max(up, up1) : up
dn = src + Multiplier * atr * sensitivity
dn1 = nz(dn[1], dn)
dn := close[1] < dn1 ? math.min(dn, dn1) : dn

// Trend determination
trend = 1
trend := nz(trend[1], trend)
trend := trend == -1 and close > dn1 ? 1 : trend == 1 and close < up1 ? -1 : trend

// Define buy and sell conditions
buySignal = trend == 1 and trend[1] == -1
sellSignal = trend == -1 and trend[1] == 1

// Entry conditions
if buySignal and strategy.position_size == 0
    strategy.entry('BUY', strategy.long)
if sellSignal and strategy.position_size == 0
    strategy.entry('SELL', strategy.short)

// Implement trailing stop loss
strategy.exit('Exit Long', 'BUY', trail_points = atr * trailingStop / 100, trail_offset = atr * trailingStop / 100)
strategy.exit('Exit Short', 'SELL', trail_points = atr * trailingStop / 100, trail_offset = atr * trailingStop / 100)
