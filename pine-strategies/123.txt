//@version=6
indicator("Engulfing Indicator with Trailing TP/SL for Bybit Webhook", overlay=true)

// === INPUTS ===
// Time Window Inputs
useTimeFilter = input.bool(true, title="Enable Time Filter?", group="Time Filter")
FromMonth = input.int(1, title="From Month", minval=1, maxval=12, group="Time Filter")
FromDay = input.int(1, title="From Day", minval=1, maxval=31, group="Time Filter")
FromYear = input.int(2017, title="From Year", minval=2017, group="Time Filter")
ToMonth = input.int(12, title="To Month", minval=1, maxval=12, group="Time Filter")
ToDay = input.int(31, title="To Day", minval=1, maxval=31, group="Time Filter")
ToYear = input.int(9999, title="To Year", minval=2017, group="Time Filter")

// Order Size and TP/SL Inputs
orderSize = input.float(1.0, title="Order Size (Contracts)", minval=0.1, maxval=100.0, step=0.1, group="Order Settings")
takeProfitPercent = input.float(0.8, title="Hard Take-Profit (%)", minval=0.1, maxval=10.0, step=0.1, group="Hard TP/SL")
stopLossPercent = input.float(0.8, title="Hard Stop-Loss (%)", minval=0.1, maxval=10.0, step=0.1, group="Hard TP/SL")
trailingTakeProfitPercent = input.float(0.7, title="Trailing Take-Profit (%)", minval=0.1, maxval=10.0, step=0.1, group="Trailing TP/SL")
trailingStopLossPercent = input.float(0.8, title="Trailing Stop-Loss (%)", minval=0.1, maxval=10.0, step=0.1, group="Trailing TP/SL")
lookbackPeriod = input.int(10, title="Trailing Lookback Period (Bars)", minval=1, maxval=50, group="Trailing TP/SL")

// === TIME FILTER ===
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
timeFilterPass = not useTimeFilter or (time >= start and time <= finish)

// === ENGULFING CANDLES LOGIC ===
openBarPrevious = open[1]
closeBarPrevious = close[1]
openBarCurrent = open
closeBarCurrent = close

bullishEngulfing = (openBarCurrent <= closeBarPrevious) and (openBarCurrent < openBarPrevious) and (closeBarCurrent > openBarPrevious)
bearishEngulfing = (openBarCurrent >= closeBarPrevious) and (openBarCurrent > openBarPrevious) and (closeBarCurrent < openBarPrevious)

// Plot signals
plotshape(bullishEngulfing, style=shape.triangleup, location=location.belowbar, color=color.green, size=size.tiny)
plotshape(bearishEngulfing, style=shape.triangledown, location=location.abovebar, color=color.red, size=size.tiny)

// === ENTRY CONDITIONS ===
finalLongCondition = bullishEngulfing and timeFilterPass
finalShortCondition = bearishEngulfing and timeFilterPass

// === TP/SL CALCULATIONS (Initial Fixed Levels for Entry) ===
longTakeProfit = close * (1 + takeProfitPercent / 100)
longStopLoss = close * (1 - stopLossPercent / 100)
shortTakeProfit = close * (1 - takeProfitPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// === TRAILING LOGIC ===
// Calculate highest/lowest prices on every bar
highestPrice = ta.highest(high, lookbackPeriod)
lowestPrice = ta.lowest(low, lookbackPeriod)

var float longEntryPrice = 0.0
var float shortEntryPrice = 0.0
var float longTrailingTP = 0.0
var float longTrailingSL = 0.0
var float shortTrailingTP = 0.0
var float shortTrailingSL = 0.0
var bool longActive = false
var bool shortActive = false

// Update entry prices and reset trailing levels on new trades
if finalLongCondition
    longEntryPrice := close
    longTrailingTP := close * (1 + trailingTakeProfitPercent / 100)
    longTrailingSL := close * (1 - trailingStopLossPercent / 100)
    longActive := true
    shortActive := false  // Close any short position
if finalShortCondition
    shortEntryPrice := close
    shortTrailingTP := close * (1 - trailingTakeProfitPercent / 100)
    shortTrailingSL := close * (1 + trailingStopLossPercent / 100)
    shortActive := true
    longActive := false  // Close any long position

// Update trailing levels for Long position
if longActive
    longTrailingTP := math.max(longTrailingTP, highestPrice * (1 - trailingTakeProfitPercent / 100))
    longTrailingSL := math.max(longTrailingSL, highestPrice * (1 - trailingStopLossPercent / 100))

// Update trailing levels for Short position
if shortActive
    shortTrailingTP := math.min(shortTrailingTP, lowestPrice * (1 + trailingTakeProfitPercent / 100))
    shortTrailingSL := math.min(shortTrailingSL, lowestPrice * (1 + trailingStopLossPercent / 100))

// Close conditions for Long position
longCloseTP = longActive and close >= longTrailingTP
longCloseSL = longActive and close <= longTrailingSL
if longCloseTP or longCloseSL
    longActive := false

// Close conditions for Short position
shortCloseTP = shortActive and close <= shortTrailingTP
shortCloseSL = shortActive and close >= shortTrailingSL
if shortCloseTP or shortCloseSL
    shortActive := false

// === ALERTS WITH WEBHOOK JSON FOR BYBIT ===
// Entry Alerts
longAlertMsg = '{"side":"Buy","symbol":"{{ticker}}","qty":' + str.tostring(orderSize) + ',"type":"Market","timeInForce":"IOC","takeProfit":' + str.tostring(longTakeProfit) + ',"stopLoss":' + str.tostring(longStopLoss) + '}'
shortAlertMsg = '{"side":"Sell","symbol":"{{ticker}}","qty":' + str.tostring(orderSize) + ',"type":"Market","timeInForce":"IOC","takeProfit":' + str.tostring(shortTakeProfit) + ',"stopLoss":' + str.tostring(shortStopLoss) + '}'
closeAlertMsg = '{"side":"Close","symbol":"{{ticker}}"}'

// Trigger entry alerts
if finalLongCondition
    alert(longAlertMsg, alert.freq_once_per_bar_close)
if finalShortCondition
    alert(shortAlertMsg, alert.freq_once_per_bar_close)

// Trigger close alerts for Long position
if longCloseTP or longCloseSL
    alert(closeAlertMsg, alert.freq_once_per_bar_close)

// Trigger close alerts for Short position
if shortCloseTP or shortCloseSL
    alert(closeAlertMsg, alert.freq_once_per_bar_close)
