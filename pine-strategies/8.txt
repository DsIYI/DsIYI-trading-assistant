//@version=5
strategy("Pivot Points and SuperTrend Strategy with Candle Confirmation", overlay=true)

// Trading settings inputs
length = input.int(2, 'Pivot Length', minval=1, maxval=50)
Factor = input.float(3, "ATR Factor", minval=1, step=0.1)
Pd = input.int(10, "ATR Period", minval=1)
stopLossPercent = input.float(1.0, 'Stop-Loss (%)', minval=0.1, maxval=10.0)

// Timing settings
FromMonth = input.int(4, 'From Month', minval=1, maxval=12)
FromDay = input.int(1, 'From Day', minval=1, maxval=31)
FromYear = input.int(2024, 'From Year', minval=2024)
ToMonth = input.int(12, 'To Month', minval=1, maxval=12)
ToDay = input.int(1, 'To Day', minval=1, maxval=31)
ToYear = input.int(9999, 'To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)

window() =>
    time >= start and time <= finish

// Calculate ATR for SuperTrend
atr = ta.atr(Pd)

// Initialize SuperTrend trend variables
var float TUp = na
var float TDown = na
TUp := nz(TUp[1], close - Factor * atr)
TDown := nz(TDown[1], close + Factor * atr)
TUp := close[1] > TUp[1] ? math.max(close - Factor * atr, TUp[1]) : close - Factor * atr
TDown := close[1] < TDown[1] ? math.min(close + Factor * atr, TDown[1]) : close + Factor * atr

// Determine the trend
var int Trend = na
Trend := close > TDown[1] ? 1 : close < TUp[1] ? -1 : nz(Trend[1], 0)

// Pivot points
ph = ta.pivothigh(length, length)
pl = ta.pivotlow(length, length)

// Conditions for entry based on pivot points and candle color within the defined window
longCondition = not na(pl) and (close > open) and window()
shortCondition = not na(ph) and (close < open) and window()

if (longCondition)
    if (strategy.position_size < 0)
        strategy.close_all(comment="Close Sell to Open Buy")
    strategy.entry("Buy Entry", strategy.long)
    strategy.exit("Exit Buy", "Buy Entry", stop=close * (1 - stopLossPercent / 100))

if (shortCondition)
    if (strategy.position_size > 0)
        strategy.close_all(comment="Close Buy to Open Sell")
    strategy.entry("Sell Entry", strategy.short)
    strategy.exit("Exit Sell", "Sell Entry", stop=close * (1 + stopLossPercent / 100))

plot(TUp, color=color.green)
plot(TDown, color=color.red)
