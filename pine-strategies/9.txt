//@version=6
strategy('Improved Linear Regression Bull and Bear Power Strategy (Heikin Ashi)', shorttitle='LRBBP_HA_Stra_Improved', overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=100)

// === Time Filter Inputs ===
FromMonth = input.int(9, title='From Month', minval=1, maxval=12)
FromDay   = input.int(1, title='From Day', minval=1, maxval=31)
FromYear  = input.int(2018, title='From Year', minval=999)
ToMonth   = input.int(1, title='To Month', minval=1, maxval=12)
ToDay     = input.int(1, title='To Day', minval=1, maxval=31)
ToYear    = input.int(9999, title='To Year', minval=999)

start  = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
in_time_window = (time >= start and time <= finish)

// === Heikin Ashi Calculation ===
var float haOpen = na
var float haClose = na
haClose := (open + high + low + close) / 4
haOpen := na(haOpen[1]) ? (open + close) / 2 : (haOpen[1] + haClose[1]) / 2
haHigh = math.max(high, math.max(haOpen, haClose))
haLow = math.min(low, math.min(haOpen, haClose))

isGreenCandle = haClose > haOpen
isRedCandle   = haClose < haOpen

// === Main Indicator Inputs ===
window = input.int(10, title='Lookback Window')
smooth = input.bool(true, title='Smooth Direction with ALMA?')
smap   = input.int(5, title='ALMA Smooth factor', minval=2, maxval=10)
sigma  = input.int(6, title='ALMA Sigma')

// === Linear Regression Bull and Bear Power ===
f_exp_lr(_height, _length) =>
    _height + _height / _length

h_value = ta.highest(close, window)
l_value = ta.lowest(close, window)
h_bar   = bar_index - ta.highestbars(close, window)
l_bar   = bar_index - ta.lowestbars(close, window)

f_exp_lr__1 = f_exp_lr(h_value - close, bar_index - h_bar)
bear = 0 - (f_exp_lr(h_value - close, bar_index - h_bar) > 0 ? f_exp_lr__1 : 0)

f_exp_lr__2 = f_exp_lr(close - l_value, bar_index - l_bar)
bull = 0 + (f_exp_lr(close - l_value, bar_index - l_bar) > 0 ? f_exp_lr__2 : 0)

rawPower = bull + bear
alma_1 = ta.alma(rawPower, smap, 0.9, sigma)
direction = smooth ? alma_1 : bull * 3 + bear * 3

// === Zero Lag EMA ===
length = input.int(10, title='Zero Lag EMA Length')
src    = close
ema1   = ta.ema(src, length)
ema2   = ta.ema(ema1, length)
d      = ema1 - ema2
zlema  = ema1 + d
plot(zlema, color=color.blue, title='ZLEMA')

// === Optional Enhancements (Toggleable) ===
enableTimeFilter     = input.bool(true, title='Enable Time Filter')
enableZLEMAFilter    = input.bool(true, title='Enable ZLEMA Filter')
enableDirectionLogic = input.bool(true, title='Enable Direction Filter')
enableColorFilter    = input.bool(true, title='Enable Candle Color Filter')
enableChopFilter     = input.bool(true, title='Enable Choppiness Filter')

// === Optional Choppiness Filter ===
chopLength = input.int(14, title='Choppiness Lookback', inline='chop')
chopThreshold = input.float(61.8, title='Max Choppiness Threshold', minval=0, maxval=100, inline='chop')

// Calculate True Range (TR) for Choppiness filter
tr = ta.tr(true)

// Calculate Sum of TR over the lookback period
sumTR = 0.0
for i = 0 to chopLength - 1
    sumTR := sumTR + tr[i]

// Calculate the Choppiness Index
range_ = ta.highest(high, chopLength) - ta.lowest(low, chopLength)
choppiness = 100 * math.log10(sumTR / range_) / math.log10(chopLength)
isNotChoppy = choppiness < chopThreshold

// === Stop-Loss Percentage Input (User Adjustable) ===
stopLossPercentage = input.float(0.001, title="Stop-Loss Percentage", minval=0.001, step=0.001) / 100  // Default: 0.001% Stop-Loss

// === Stop-Loss Calculation (0.001% accuracy) ===
longStopLossPrice = strategy.position_avg_price * (1 - stopLossPercentage)
shortStopLossPrice = strategy.position_avg_price * (1 + stopLossPercentage)

// === Strategy Logic ===
longCondition  = (not enableZLEMAFilter or haClose > zlema) and
                 (not enableDirectionLogic or (smooth ? direction[0] > direction[1] : direction > bull)) and
                 (not enableColorFilter or isGreenCandle) and
                 (not enableChopFilter or isNotChoppy)

shortCondition = (not enableZLEMAFilter or haClose < zlema) and
                 (not enableDirectionLogic or (smooth ? direction[0] < direction[1] : direction < bear)) and
                 (not enableColorFilter or isRedCandle) and
                 (not enableChopFilter or isNotChoppy)

// === Strategy Entry with Stop-Loss ===
if not enableTimeFilter or in_time_window
    if longCondition
        strategy.entry('Buy', strategy.long)
        strategy.exit('Sell Long', 'Buy', stop=longStopLossPrice)
    if shortCondition
        strategy.entry('Sell', strategy.short)
        strategy.exit('Buy Short', 'Sell', stop=shortStopLossPrice)
