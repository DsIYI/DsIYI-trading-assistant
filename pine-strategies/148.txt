//@version=6
strategy('Enhanced Pivot Point Strategy with Independent Signals', overlay = true, default_qty_type = strategy.percent_of_equity, default_qty_value = 25, max_bars_back = 4999, initial_capital = 2000)

// Timing settings
FromMonth = input.int(defval = 7, title = 'From Month', minval = 1, maxval = 12, group = 'Timing')
FromDay = input.int(defval = 5, title = 'From Day', minval = 1, maxval = 31, group = 'Timing')
FromYear = input.int(defval = 2024, title = 'From Year', minval = 2021, group = 'Timing')
ToMonth = input.int(defval = 1, title = 'To Month', minval = 1, maxval = 12, group = 'Timing')
ToDay = input.int(defval = 1, title = 'To Day', minval = 1, maxval = 31, group = 'Timing')
ToYear = input.int(defval = 9999, title = 'To Year', minval = 2024, group = 'Timing')

start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)

window() =>
    time >= start and time <= finish

// Pivot Point settings
pivotPeriod = input.int(defval = 30, title = 'Pivot Period', minval = 1, maxval = 50, group = 'Pivot Points')
candleWait = input.int(defval = 1, title = 'Candle Wait Period', options = [1, 2, 3, 5], group = 'Pivot Points')

// EMA and Volume Filter settings
emaLength = input.int(defval = 6, title = 'EMA Length', minval = 1, maxval = 100, group = 'EMA & Volume Filter')
volumeLookback = input.int(defval = 16, title = 'Volume Lookback Period', minval = 1, maxval = 100, group = 'EMA & Volume Filter')

// Additional Filter settings
rsiLength = input.int(defval = 14, title = 'RSI Length', minval = 1, maxval = 50, group = 'RSI Filter')
macdFastLength = input.int(defval = 12, title = 'MACD Fast Length', minval = 1, maxval = 50, group = 'MACD Filter')
macdSlowLength = input.int(defval = 26, title = 'MACD Slow Length', minval = 1, maxval = 50, group = 'MACD Filter')
macdSignalSmoothing = input.int(defval = 9, title = 'MACD Signal Smoothing', minval = 1, maxval = 50, group = 'MACD Filter')

// EMA and Volume calculation
emaValue = ta.ema(close, emaLength)
avgVolume = ta.sma(volume, volumeLookback)

// Additional filter calculations
rsiValue = ta.rsi(close, rsiLength)
[macdLine, signalLine, _] = ta.macd(close, macdFastLength, macdSlowLength, macdSignalSmoothing)

// Entry conditions for EMA
longConditionEMA = close > emaValue and volume > avgVolume and window()
shortConditionEMA = close < emaValue and volume > avgVolume and window()

// Entry conditions for RSI
longConditionRSI = rsiValue < 30 and window()
shortConditionRSI = rsiValue > 70 and window()

// Entry conditions for MACD
longConditionMACD = macdLine > signalLine and window()
shortConditionMACD = macdLine < signalLine and window()

// Stop-Loss and Trailing Stop settings
stopLossPercent = input.float(1.0, 'Stop-Loss (%)', minval = 0.1, maxval = 10.0, step = 0.1, group = 'Stop-Loss & Trailing Stop')
trailingStop = input.float(1.5, 'Trailing Stop (%)', step = 0.1, group = 'Stop-Loss & Trailing Stop')

// Calculate stop-loss levels
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// Execute EMA trades
if longConditionEMA
    strategy.close('EMA Short', comment = 'Close Short on Long EMA')
    strategy.entry('EMA Long', strategy.long)
    strategy.exit('Close Long EMA', 'EMA Long', stop = longStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

if shortConditionEMA
    strategy.close('EMA Long', comment = 'Close Long on Short EMA')
    strategy.entry('EMA Short', strategy.short)
    strategy.exit('Close Short EMA', 'EMA Short', stop = shortStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

// Execute RSI trades
if longConditionRSI
    strategy.close('RSI Short', comment = 'Close Short on Long RSI')
    strategy.entry('RSI Long', strategy.long)
    strategy.exit('Close Long RSI', 'RSI Long', stop = longStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

if shortConditionRSI
    strategy.close('RSI Long', comment = 'Close Long on Short RSI')
    strategy.entry('RSI Short', strategy.short)
    strategy.exit('Close Short RSI', 'RSI Short', stop = shortStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

// Execute MACD trades
if longConditionMACD
    strategy.close('MACD Short', comment = 'Close Short on Long MACD')
    strategy.entry('MACD Long', strategy.long)
    strategy.exit('Close Long MACD', 'MACD Long', stop = longStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

if shortConditionMACD
    strategy.close('MACD Long', comment = 'Close Long on Short MACD')
    strategy.entry('MACD Short', strategy.short)
    strategy.exit('Close Short MACD', 'MACD Short', stop = shortStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)
