//@version=6
strategy('Enhanced BB Strategy with Multi-TF EMA & Volume', overlay = true, pyramiding = 0)

// Timing settings
FromMonth = input.int(defval = 4, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(defval = 17, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(defval = 2024, title = 'From Year', minval = 2024)
ToMonth = input.int(defval = 1, title = 'To Month', minval = 1, maxval = 12)
ToDay = input.int(defval = 1, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(defval = 9999, title = 'To Year', minval = 2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish ? true : false

// Define input parameters
length = input.int(20, 'SMA Length')
mult = input.float(2.0, 'Standard Deviation Multiplier')
trailingPercent = input.float(1.5, 'Trailing Stop (%)')
pyramidMax = input.int(2, 'Max Pyramid Orders')
percentPerOrder = input.float(10, 'Percent of Equity per Order')
useEMAFilter = input.string('Both', 'Use EMA Filter for', options = ['None', '2m', '5m', 'Both'])
useVolumeFilter = input.string('All', 'Use Volume Filter for', options = ['None', '2m', '5m', 'Current', 'All'])
emaLength = 9
volumeLookback = 50 // Lookback period for average volume calculation
useDynamicTakeProfit = input.bool(true, 'Use Dynamic Take Profit?')

// Fetch EMA from different timeframes without lookahead
ema2m = request.security(syminfo.tickerid, '2', ta.ema(close, emaLength), barmerge.gaps_off, barmerge.lookahead_off)
ema5m = request.security(syminfo.tickerid, '5', ta.ema(close, emaLength), barmerge.gaps_off, barmerge.lookahead_off)

// Fetch volume from different timeframes and calculate the average volume
volume2m = request.security(syminfo.tickerid, '2', volume, barmerge.gaps_off, barmerge.lookahead_off)
volume5m = request.security(syminfo.tickerid, '5', volume, barmerge.gaps_off, barmerge.lookahead_off)
avgVolume2m = ta.sma(volume2m, volumeLookback)
avgVolume5m = ta.sma(volume5m, volumeLookback)
avgVolumeCurrent = ta.sma(volume, volumeLookback)

// Bollinger Bands calculation
basis = ta.sma(close, length)
dev = mult * ta.stdev(close, length)
upper = basis + dev
lower = basis - dev

// Entry conditions based on EMA and volume filters
longCondition = close > lower and (useEMAFilter == '2m' and close > ema2m or useEMAFilter == '5m' and close > ema5m or useEMAFilter == 'Both' and close > ema2m and close > ema5m)
shortCondition = close < upper and (useEMAFilter == '2m' and close < ema2m or useEMAFilter == '5m' and close < ema5m or useEMAFilter == 'Both' and close < ema2m and close < ema5m)
longVolume = volume > avgVolumeCurrent and (useVolumeFilter == '2m' and volume2m > avgVolume2m or useVolumeFilter == '5m' and volume5m > avgVolume5m or useVolumeFilter == 'All')
shortVolume = volume > avgVolumeCurrent and (useVolumeFilter == '2m' and volume2m > avgVolume2m or useVolumeFilter == '5m' and volume5m > avgVolume5m or useVolumeFilter == 'All')

// Combine conditions for final entry signals
finalLongCondition = longCondition and longVolume
finalShortCondition = shortCondition and shortVolume

// Pre-calculate ATR for consistency (fixes warning)
atr = ta.atr(14)

// Execute trades only within time window
if window()
    if finalLongCondition
        strategy.entry('Long Entry', strategy.long)
    if finalShortCondition
        strategy.entry('Short Entry', strategy.short)

    // Use precomputed ATR for take profit
    if useDynamicTakeProfit
        strategy.exit('TakeProfitLong', 'Long Entry', limit = close + atr * 1.5)
        strategy.exit('TakeProfitShort', 'Short Entry', limit = close - atr * 1.5)
