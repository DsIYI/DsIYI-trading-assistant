strategy("AI Generated Supertrend-RSI Strategy", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=100)

// === Inputs
atrPeriod = input.int(10, "ATR Length", minval=1)
factor = input.float(3.0, "Supertrend Factor", step=0.1)
rsiPeriod = input.int(14, "RSI Length")
rsiBuy = input.int(35, "RSI Buy Threshold")
rsiSell = input.int(65, "RSI Sell Threshold")
slPerc = input.float(1.5, "Stop-Loss %", step=0.1)
tpPerc = input.float(2.0, "Take-Profit %", step=0.1)

// === Supertrend Logic
[supertrend, direction] = ta.supertrend(factor, atrPeriod)
buySignal = ta.crossover(close, supertrend)
sellSignal = ta.crossunder(close, supertrend)

// === RSI Filter
rsi = ta.rsi(close, rsiPeriod)
longFilter = rsi < rsiBuy
shortFilter = rsi > rsiSell

// === Entries
longCondition = buySignal and longFilter
shortCondition = sellSignal and shortFilter

if (longCondition)
    strategy.entry("Long", strategy.long)
if (shortCondition)
    strategy.entry("Short", strategy.short)

// === Exits
strategy.exit("Exit Long", from_entry="Long", loss=slPerc*10, profit=tpPerc*10)
strategy.exit("Exit Short", from_entry="Short", loss=slPerc*10, profit=tpPerc*10)

// === Plot
plotshape(longCondition, title="Buy Signal", location=location.belowbar, color=color.green, style=shape.labelup, text="BUY")
plotshape(shortCondition, title="Sell Signal", location=location.abovebar, color=color.red, style=shape.labeldown, text="SELL")
