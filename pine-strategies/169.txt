//@version=6
strategy("Improved Line Break Strategy with Intrabar Signals + Clean Visuals", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=100)

// === Inputs ===
emaShortLength     = input.int(9, "Short EMA Length", minval=1)
emaLongLength      = input.int(21, "Long EMA Length", minval=1)
useRSI             = input.bool(true, "Use RSI Filter")
rsiLength          = input.int(14, "RSI Length", minval=1)
rsiOverbought      = input.int(70, "RSI Overbought Level", minval=50, maxval=100)
rsiOversold        = input.int(30, "RSI Oversold Level", minval=0, maxval=50)
stoploss_percent   = input.float(1.0, "Stop-Loss %", minval=0.1, step=0.1) / 100
trailing_tp_percent= input.float(1.5, "Trailing Take Profit %", minval=0.1, step=0.1) / 100
atrMultiplier      = input.float(1.5, "ATR Stop-Loss Multiplier", minval=0.1, step=0.1)

// === Backtest Window ===
FromMonth = input.int(1, "From Month", minval=1, maxval=12)
FromDay   = input.int(1, "From Day", minval=1, maxval=31)
FromYear  = input.int(2025, "From Year", minval=2025)
ToMonth   = input.int(12, "To Month", minval=1, maxval=12)
ToDay     = input.int(31, "To Day", minval=1, maxval=31)
ToYear    = input.int(2999, "To Year", minval=2025)

start  = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() => time >= start and time <= finish

// === Indicator Calculations ===
emaShort = ta.ema(close, emaShortLength)
emaLong  = ta.ema(close, emaLongLength)
rsiValue = ta.rsi(close, rsiLength)
atr      = ta.atr(14)
atrStop  = atr * atrMultiplier

// === Intrabar 1s Signals ===
oneSecClose = request.security(syminfo.tickerid, "1S", close)
oneSecRSI   = request.security(syminfo.tickerid, "1S", ta.rsi(close, rsiLength))
oneSecEMAS  = request.security(syminfo.tickerid, "1S", ta.ema(close, emaShortLength))
oneSecEMAL  = request.security(syminfo.tickerid, "1S", ta.ema(close, emaLongLength))

long_condition  = oneSecClose > oneSecEMAS and oneSecEMAS > oneSecEMAL and (not useRSI or oneSecRSI < rsiOversold)
short_condition = oneSecClose < oneSecEMAS and oneSecEMAS < oneSecEMAL and (not useRSI or oneSecRSI > rsiOverbought)

// === Entry Triggers (Reversible Trades) ===
if (long_condition and window())
    strategy.close("Short")
    strategy.entry("Long", strategy.long)
    strategy.exit("Exit Long", from_entry="Long", trail_offset=trailing_tp_percent * close, stop=close - atrStop)

if (short_condition and window())
    strategy.close("Long")
    strategy.entry("Short", strategy.short)
    strategy.exit("Exit Short", from_entry="Short", trail_offset=trailing_tp_percent * close, stop=close + atrStop)

// === Clean Visual Plotshapes (1st Bar Only) ===
isNewLong  = long_condition and not long_condition[1]
isNewShort = short_condition and not short_condition[1]

plotshape(series=isNewLong, title="Long Signal", location=location.belowbar, color=color.green, style=shape.labelup, text="BUY")
plotshape(series=isNewShort, title="Short Signal", location=location.abovebar, color=color.red, style=shape.labeldown, text="SELL")

// === Alert Conditions (Webhook Ready) ===
alertcondition(isNewLong, title="1s Long Signal", message="Buy Signal (1s Confirmed)")
alertcondition(isNewShort, title="1s Short Signal", message="Sell Signal (1s Confirmed)")

// === Debug EMA Plotting ===
plot(emaShort, title="Short EMA", color=color.green)
plot(emaLong, title="Long EMA", color=color.red)
