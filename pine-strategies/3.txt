//@version=5
strategy("Improved Pivot Point Strategy v1.0 - Repaint-Free & Webhook", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15, max_bars_back=4999)

// Timing settings
FromMonth = input.int(defval=4, title='From Month', minval=1, maxval=12)
FromDay = input.int(defval=17, title='From Day', minval=1, maxval=31)
FromYear = input.int(defval=2024, title='From Year', minval=2024)
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12)
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31)
ToYear = input.int(defval=9999, title='To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time[1] >= start and time[1] <= finish  // FIXED: [1] confirmed

// Pivot Point settings
pivotPeriod = input.int(defval=5, title="Pivot Period", minval=1, maxval=50)
candleWait = input.int(defval=1, title="Candle Wait Period", options=[1, 2, 3, 5])

// EMA and Volume Filter settings
emaLength = 9
volumeLookback = 50

// EMA and Volume calculation (historical [1] for no repaint)
emaValue = ta.ema(close[1], emaLength)[1]  // FIXED: [1] confirmed
avgVolume = ta.sma(volume[1], volumeLookback)[1]  // FIXED: [1]

// Pivot Point calculations (historical)
pivotHigh = ta.pivothigh(high[1], pivotPeriod, pivotPeriod)[1]  // FIXED: [1] confirmed
pivotLow = ta.pivotlow(low[1], pivotPeriod, pivotPeriod)[1]  // FIXED: [1]

// Plot Pivot Points
plotshape(series=pivotHigh, location=location.abovebar, color=color.red, style=shape.labeldown, title="Pivot High")
plotshape(series=pivotLow, location=location.belowbar, color=color.green, style=shape.labelup, title="Pivot Low")

// Check if Pivot Points have stayed for the specified candleWait period
var int barsSincePivotHigh = na
var int barsSincePivotLow = na
barsSincePivotHigh := na(pivotHigh[1]) ? (na(barsSincePivotHigh) ? 0 : barsSincePivotHigh + 1) : 0  // FIXED: [1]
barsSincePivotLow := na(pivotLow[1]) ? (na(barsSincePivotLow) ? 0 : barsSincePivotLow + 1) : 0  // FIXED: [1]

pivotHighCondition = barsSincePivotHigh[1] >= candleWait  // FIXED: [1]
pivotLowCondition = barsSincePivotLow[1] >= candleWait  // FIXED: [1]

// Entry conditions based on Pivot Points, EMA, and volume (confirmed)
longConditionPivot = barstate.isconfirmed and pivotLowCondition and window()  // FIXED: Confirmed
shortConditionPivot = barstate.isconfirmed and pivotHighCondition and window()  // FIXED: Confirmed

longConditionEMA = barstate.isconfirmed and (close[1] > emaValue) and (volume[1] > avgVolume) and window()  // FIXED: [1] + Confirmed
shortConditionEMA = barstate.isconfirmed and (close[1] < emaValue) and (volume[1] > avgVolume) and window()  // FIXED: [1] + Confirmed

// Slippage and Spread consideration
slippage = input.float(defval=0.1, title="Slippage (%)")
spread = input.float(defval=0.1, title="Spread (%)")

// Stop-Loss and Trailing Stop settings
stopLossPercent = input.float(1.0, "Stop-Loss (%)", minval=0.1, maxval=10.0) + slippage + spread
trailingStop = input.float(1.5, "Trailing Stop (%)", step=0.1) + slippage + spread

// Calculate stop-loss levels (from confirmed entry)
var float longEntry = na
var float shortEntry = na
longStopLoss = na(longEntry) ? na : longEntry * (1 - stopLossPercent / 100)
shortStopLoss = na(shortEntry) ? na : shortEntry * (1 + stopLossPercent / 100)

// Execute trades with reversible signals and regular stop-loss
if (longConditionPivot or longConditionEMA)
    strategy.close("Short Entry", comment="Close Short on Long Signal") // Close existing short positions
    if (longConditionPivot)
        strategy.entry("Long Entry Pivot", strategy.long)
        longEntry := close[1]  // FIXED: Confirmed entry
        strategy.exit("Close Long Pivot", "Long Entry Pivot", stop=longStopLoss, trail_points=trailingStop / syminfo.mintick, trail_offset=trailingStop / syminfo.mintick)
    if (longConditionEMA)
        strategy.entry("Long Entry EMA", strategy.long)
        longEntry := close[1]  // FIXED: Confirmed entry
        strategy.exit("Close Long EMA", "Long Entry EMA", stop=longStopLoss, trail_points=trailingStop / syminfo.mintick, trail_offset=trailingStop / syminfo.mintick)

if (shortConditionPivot or shortConditionEMA)
    strategy.close("Long Entry", comment="Close Long on Short Signal") // Close existing long positions
    if (shortConditionPivot)
        strategy.entry("Short Entry Pivot", strategy.short)
        shortEntry := close[1]  // FIXED: Confirmed entry
        strategy.exit("Close Short Pivot", "Short Entry Pivot", stop=shortStopLoss, trail_points=trailingStop / syminfo.mintick, trail_offset=trailingStop / syminfo.mintick)
    if (shortConditionEMA)
        strategy.entry("Short Entry EMA", strategy.short)
        shortEntry := close[1]  // FIXED: Confirmed entry
        strategy.exit("Close Short EMA", "Short Entry EMA", stop=shortStopLoss, trail_points=trailingStop / syminfo.mintick, trail_offset=trailingStop / syminfo.mintick)

// Reset entries on close
if strategy.position_size == 0
    longEntry := na
    shortEntry := na

// Debugging: Plot conditions (confirmed)
plotshape(series=longConditionPivot, title="Long Condition Pivot", location=location.belowbar, color=color.blue, style=shape.labelup, size=size.small)
plotshape(series=shortConditionPivot, title="Short Condition Pivot", location=location.abovebar, color=color.orange, style=shape.labeldown, size=size.small)

plotshape(series=longConditionEMA, title="Long Condition EMA", location=location.belowbar, color=color.green, style=shape.labelup, size=size.small)
plotshape(series=shortConditionEMA, title="Short Condition EMA", location=location.abovebar, color=color.red, style=shape.labeldown, size=size.small)

// NEW: Webhook Alerts (Separate for Entry/Exit)
if longConditionPivot or longConditionEMA
    alert('{"symbol":"TAOUSDT.P","side":"buy","qty":"' + str.tostring(math.round(strategy.position_size * 0.01, 3)) + '","price":"' + str.tostring(close[1]) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

if shortConditionPivot or shortConditionEMA
    alert('{"symbol":"TAOUSDT.P","side":"sell","qty":"' + str.tostring(math.round(math.abs(strategy.position_size) * 0.01, 3)) + '","price":"' + str.tostring(close[1]) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

// Exit Alerts (Reduce Only for Closes - Trigger on Opposite Signal)
if strategy.position_size > 0 and (shortConditionPivot or shortConditionEMA)
    alert('{"symbol":"TAOUSDT.P","side":"sell","qty":"' + str.tostring(math.round(strategy.position_size, 3)) + '","price":"' + str.tostring(close[1]) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

if strategy.position_size < 0 and (longConditionPivot or longConditionEMA)
    alert('{"symbol":"TAOUSDT.P","side":"buy","qty":"' + str.tostring(math.round(math.abs(strategy.position_size), 3)) + '","price":"' + str.tostring(close[1]) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)
