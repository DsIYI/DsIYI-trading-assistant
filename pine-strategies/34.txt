//@version=5
strategy('Candlestick Trend Strategy v0.5.3 by JustUncleL - Scope-Fixed + Green Compile', overlay=true, max_bars_back=500, calc_on_every_tick=false)

// Indicator Inputs (tuned for MYX vol)
matype   = input.string(defval='HullMA', title='Fast MA Type', options=['SMA', 'EMA', 'WMA', 'VWMA', 'SMMA', 'DEMA', 'TEMA', 'HullMA', 'TMA', 'ZEMA'])
malength = input.int(defval=14, title='Moving Average Length', minval=1)  // OPT: 14 for MYX responsiveness
src      = input.source(close, title='Moving Average Source')  // FIXED: Plain close

// Tuned Filters (Loosened for Better Perf)
useVolFilter = input.bool(true, title="Volume Filter (Vol > 1.3x Avg)?")  // Tuned: 1.3x (was 1.5x)
useTrendFilter = input.bool(true, title="Trend Filter (ADX > 18)?")  // Tuned: 18 (was 20)
adxLen = input.int(14, title="ADX Length")
minAdx = input.float(18, title="Min ADX")  // Tuned for more signals
signalCooldown = input.int(1, title="Cooldown Bars Between Signals", minval=1)  // Tuned: 1 (was 2)
var int lastSignalBar = na

// FIXED: Cooldown calc (early, historical)
cooldownPassed = na(lastSignalBar) or bar_index - lastSignalBar > signalCooldown

// Backtesting Time Settings (live-friendly)
fromMonth = input.int(1, title='From Month', minval=1, maxval=12)
fromDay = input.int(1, title='From Day', minval=1, maxval=31)
fromYear = input.int(2020, title='From Year', minval=2000)
toMonth = input.int(12, title='To Month', minval=1, maxval=12)
toDay = input.int(31, title='To Day', minval=1, maxval=31)
toYear = input.int(2025, title='To Year', minval=2000)
startTime = timestamp(fromYear, fromMonth, fromDay, 00, 00)
endTime = timestamp(toYear, toMonth, toDay, 23, 59)
tradeWindow = time >= startTime and time <= endTime

// Stop-Loss & Trailing Take-Profit Inputs (tuned for MYX)
stopLossATR = input.float(1.0, title='ATR Multiplier for SL', minval=0.5, step=0.1)
trailingTPPercent = input.float(2.5, title='Trailing Take-Profit %', minval=0.1) / 100
breakevenTrigger = input.float(0.5, title="Breakeven % Before Trail", minval=0.1) / 100

// Candlestick Pattern Inputs
useBullishEngulfing = input.bool(title='Use Bullish Engulfing', defval=true)
useBearishEngulfing = input.bool(title='Use Bearish Engulfing', defval=true)
useBullishPinBar = input.bool(title='Use Bullish Pin Bar', defval=true)
useBearishPinBar = input.bool(title='Use Bearish Pin Bar', defval=true)
usePiercingLine = input.bool(title='Use Piercing Line', defval=false)
useDarkCloudCover = input.bool(title='Use Dark Cloud Cover', defval=false)

// NEW: MA Trend Alignment (for better entries)
useMATrend = input.bool(true, title="Require MA Trend Alignment?")

// Filters (all historical [1]/[2])
atrValue = ta.atr(14)[1]  // FIXED: [1] confirmed
volFilter = not useVolFilter or volume[2] > ta.sma(volume[2], 20) * 1.3  // FIXED: [2] for full confirm
[diplus, diminus, adxValue] = ta.dmi(adxLen, adxLen)
trendFilter = not useTrendFilter or adxValue[1] > minAdx  // FIXED: [1]

// Moving Average Selection (complete, historical)
variant(type, src, len) =>
    v1 = ta.sma(src, len)
    v2 = ta.ema(src, len)
    v3 = ta.wma(src, len)
    v4 = ta.vwma(src, len)
    v5 = ta.rma(src, len)  // SMMA
    v6 = 2 * v2 - ta.ema(v2, len)  // DEMA
    v7 = 3 * (v2 - ta.ema(v2, len)) + ta.ema(ta.ema(v2, len), len)  // TEMA
    v8 = ta.wma(2 * ta.wma(src, len / 2) - ta.wma(src, len), math.round(math.sqrt(len)))  // HullMA
    ema1 = ta.ema(src, len)
    ema2 = ta.ema(ema1, len)
    v10 = ema1 + (ema1 - ema2)  // ZLEMA
    v11 = ta.sma(ta.sma(src, len), len)  // TMA
    switch type
        'EMA' => v2
        'WMA' => v3
        'VWMA' => v4
        'SMMA' => v5
        'DEMA' => v6
        'TEMA' => v7
        'HullMA' => v8
        'ZEMA' => v10
        'TMA' => v11
        => v1

maValue = variant(matype, src[1], malength)[1]  // FIXED: Double [1] for no repaint
maTrendUp = maValue > maValue[1]  // Historical trend
col = maTrendUp ? color.green : color.red
plot(maValue, color=col, linewidth=2, title='MA')

// Candlestick Patterns (FULL [2]/[3] refs - zero repaint)
bullishEngulfing = useBullishEngulfing and close[3] < open[3] and close[2] > open[2] and close[2] >= high[3] and open[2] <= close[3]
bearishEngulfing = useBearishEngulfing and close[3] > open[3] and close[2] < open[2] and close[2] <= low[3] and open[2] >= close[3]

bullishPinBar = useBullishPinBar and low[2] == math.min(low[2], open[2], close[2]) and (high[2] - low[2]) > 0 and (close[2] - low[2]) / (high[2] - low[2]) > 0.66 and close[2] > open[2]
bearishPinBar = useBearishPinBar and high[2] == math.max(high[2], open[2], close[2]) and (high[2] - low[2]) > 0 and (high[2] - close[2]) / (high[2] - low[2]) > 0.66 and close[2] < open[2]

piercingLine = usePiercingLine and open[3] > close[4] and close[3] > open[3] and close[3] > (open[4] + close[4]) / 2 and close[3] < open[4]
darkCloudCover = useDarkCloudCover and open[3] < close[4] and close[3] < open[3] and close[3] < (open[4] + close[4]) / 2 and close[3] > close[4]

// FIXED: Global Signal Declarations (scope-safe)
bullishSignal = false
bearishSignal = false

// FIXED: Conditions (confirmed + historical + MA align + no same-bar reverse)
bullishCondition = (bullishEngulfing or bullishPinBar or piercingLine) and (not useMATrend or maTrendUp[1])
bearishCondition = (bearishEngulfing or bearishPinBar or darkCloudCover) and (not useMATrend or not maTrendUp[1])

var bool blockBullish = false
var bool blockBearish = false

if barstate.isconfirmed
    tempBullish = bullishCondition and volFilter and trendFilter and tradeWindow and cooldownPassed and not blockBullish
    tempBearish = bearishCondition and volFilter and trendFilter and tradeWindow and cooldownPassed and not blockBearish
    
    // Block same-bar opposites
    if tempBullish and tempBearish
        tempBullish := false
        tempBearish := false
    
    // Assign to globals
    bullishSignal := tempBullish
    bearishSignal := tempBearish
    
    // Update cooldown
    if bullishSignal or bearishSignal
        lastSignalBar := bar_index

// Track entries (historical)
var float longEntry = na
var float shortEntry = na

// FIXED: Persistent Flags for NEXT-BAR Closes (ensures distinct times)
var bool pendingCloseLong = false
var bool pendingCloseShort = false

// Entries (only on confirmed signal, no position)
if bullishSignal and strategy.position_size == 0
    strategy.entry('BUY', strategy.long)
    longEntry := close[1]  // Confirmed entry ref
    if strategy.position_size < 0
        pendingCloseShort := true  // Flag reversal for next bar

if bearishSignal and strategy.position_size == 0
    strategy.entry('SELL', strategy.short)
    shortEntry := close[1]  // Confirmed entry ref
    if strategy.position_size > 0
        pendingCloseLong := true  // Flag reversal for next bar

// FIXED: Delayed Closes (NEXT confirmed bar ONLY - no same-time)
if pendingCloseLong and barstate.isconfirmed
    strategy.close('BUY', comment="Delayed Close Long")
    pendingCloseLong := false
    longEntry := na

if pendingCloseShort and barstate.isconfirmed
    strategy.close('SELL', comment="Delayed Close Short")
    pendingCloseShort := false
    shortEntry := na

// ENHANCED: Dynamic ATR SL + Trailing (historical refs, precise offset)
longBreakeven = strategy.position_size > 0 and not na(longEntry) and (close - longEntry) / longEntry >= breakevenTrigger
shortBreakeven = strategy.position_size < 0 and not na(shortEntry) and (shortEntry - close) / shortEntry >= breakevenTrigger

if strategy.position_size > 0 and not na(longEntry)
    sl_long = longEntry - (atrValue * stopLossATR)
    if longBreakeven
        strategy.exit('Exit Long', 'BUY', stop=sl_long, trail_price=close, trail_offset=math.round(trailingTPPercent * close / syminfo.mintick))
    else
        strategy.exit('Exit Long', 'BUY', stop=sl_long)

if strategy.position_size < 0 and not na(shortEntry)
    sl_short = shortEntry + (atrValue * stopLossATR)
    if shortBreakeven
        strategy.exit('Exit Short', 'SELL', stop=sl_short, trail_price=close, trail_offset=math.round(trailingTPPercent * close / syminfo.mintick))
    else
        strategy.exit('Exit Short', 'SELL', stop=sl_short)

// Visualization (now safe - globals in scope)
plotshape(series=bullishSignal, location=location.belowbar, color=color.green, style=shape.labelup, size=size.small, title='Bullish Signal')
plotshape(series=bearishSignal, location=location.abovebar, color=color.red, style=shape.labeldown, size=size.small, title='Bearish Signal')

// Debug Table (enhanced)
if barstate.islast
    var table debugTable = table.new(position.top_right, 2, 6, bgcolor=color.white, border_width=1)
    table.cell(debugTable, 0, 0, "Vol Filter", text_color=color.black)
    table.cell(debugTable, 1, 0, volFilter ? "PASS" : "FAIL", text_color=volFilter ? color.green : color.red)
    table.cell(debugTable, 0, 1, "Trend Filter", text_color=color.black)
    table.cell(debugTable, 1, 1, trendFilter ? "PASS" : "FAIL", text_color=trendFilter ? color.green : color.red)
    table.cell(debugTable, 0, 2, "Cooldown", text_color=color.black)
    table.cell(debugTable, 1, 2, cooldownPassed ? "OK" : str.tostring(bar_index - lastSignalBar), text_color=cooldownPassed ? color.green : color.red)
    table.cell(debugTable, 0, 3, "ADX", text_color=color.black)
    table.cell(debugTable, 1, 3, str.tostring(math.round(adxValue[1])), text_color=adxValue[1] > minAdx ? color.green : color.red)
    table.cell(debugTable, 0, 4, "MA Trend", text_color=color.black)
    table.cell(debugTable, 1, 4, maTrendUp[1] ? "UP" : "DOWN", text_color=maTrendUp[1] ? color.green : color.red)
    table.cell(debugTable, 0, 5, "Position", text_color=color.black)
    table.cell(debugTable, 1, 5, strategy.position_size > 0 ? "LONG" : strategy.position_size < 0 ? "SHORT" : "FLAT", text_color=strategy.position_size != 0 ? color.blue : color.gray)
