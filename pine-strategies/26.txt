//@version=6
strategy('Reversal Strategy with Volume Delta', overlay = true)

//=============================================================================
// 1) Time Window Filters
//=============================================================================
useTimeFilter = input.bool(true, title = 'Enable Time Filter?')
FromMonth = input.int(4, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(17, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(2024, title = 'From Year', minval = 2024)
ToMonth = input.int(12, title = 'To Month', minval = 1, maxval = 12)
ToDay = input.int(31, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(9999, title = 'To Year', minval = 2024)

start = timestamp(FromYear, FromMonth, FromDay, 0, 0)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
timeFilterPass = not useTimeFilter or time >= start and time <= finish

// Calculate volume delta
upAndDownVolume() =>
    posVol = 0.0
    negVol = 0.0

    var isBuyVolume = true

    switch 
        close > open => 
    	    isBuyVolume := true
    	    isBuyVolume
        close < open => 
    	    isBuyVolume := false
    	    isBuyVolume
        close > close[1] => 
    	    isBuyVolume := true
    	    isBuyVolume
        close < close[1] => 
    	    isBuyVolume := false
    	    isBuyVolume

    if isBuyVolume
        posVol := posVol + volume
        posVol
    else
        negVol := negVol - volume
        negVol

    posVol + negVol

volumeDelta = upAndDownVolume()

// Buy Signal
buySignal = close > open and volumeDelta > 0
plotshape(buySignal, style = shape.triangleup, location = location.belowbar, color = color.green, size = size.small, title = 'Buy Signal')

// Sell Signal
sellSignal = close < open and volumeDelta < 0
plotshape(sellSignal, style = shape.triangledown, location = location.abovebar, color = color.red, size = size.small, title = 'Sell Signal')

// Define stop-loss prices and take-profit percentages
var float buySlPrice = na
var float sellSlPrice = na
var float takeProfitBuy = 1.0
var float takeProfitSell = 1.0

// Entry and Exit logic
if buySignal
    strategy.entry('Buy', strategy.long)
    strategy.exit('Take Profit Buy', 'Buy', profit = takeProfitBuy)

if sellSignal
    strategy.entry('Sell', strategy.short)
    strategy.exit('Take Profit Sell', 'Sell', profit = takeProfitSell)

strategy.exit('Stop Loss Buy', 'Buy', stop = buySlPrice)
strategy.exit('Stop Loss Sell', 'Sell', stop = sellSlPrice)

// Plot volume delta
plot(volumeDelta, title = 'Volume Delta', color = volumeDelta > 0 ? color.green : color.red, style = plot.style_histogram)
hline(0, 'Zero Line', color = color.gray)
