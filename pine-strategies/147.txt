//@version=5
strategy("Enhanced Pivot Point Strategy with EMA, Volume, RSI & MACD Filters, Reversal, and Trailing Stop", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15, max_bars_back=4999)

// Timing settings with manual input
FromMonth = input.int(defval=7, title='From Month', minval=1, maxval=12, group="Timing")
FromDay = input.int(defval=5, title='From Day', minval=1, maxval=31, group="Timing")
FromYear = input.int(defval=2024, title='From Year', minval=2024, group="Timing")
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12, group="Timing")
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31, group="Timing")
ToYear = input.int(defval=9999, title='To Year', minval=2024, group="Timing")

start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)

window() =>
    time >= start and time <= finish

// Pivot Point settings with manual input
pivotPeriod = input.int(defval=5, title="Pivot Period", minval=1, maxval=50, group="Pivot Points")
candleWait = input.int(defval=1, title="Candle Wait Period", options=[1, 2, 3, 5], group="Pivot Points")

// EMA and Volume Filter settings with manual input
emaLength = input.int(defval=9, title="EMA Length", minval=1, maxval=100, group="EMA & Volume Filter")
volumeLookback = input.int(defval=50, title="Volume Lookback Period", minval=1, maxval=100, group="EMA & Volume Filter")

// Additional Filter settings with manual input
rsiLength = input.int(defval=14, title="RSI Length", minval=1, maxval=50, group="RSI Filter")
macdFastLength = input.int(defval=12, title="MACD Fast Length", minval=1, maxval=50, group="MACD Filter")
macdSlowLength = input.int(defval=26, title="MACD Slow Length", minval=1, maxval=50, group="MACD Filter")
macdSignalSmoothing = input.int(defval=9, title="MACD Signal Smoothing", minval=1, maxval=50, group="MACD Filter")

// EMA and Volume calculation
emaValue = ta.ema(close, emaLength)
avgVolume = ta.sma(volume, volumeLookback)

// Pivot Point calculations
pivotHigh = ta.pivothigh(high, pivotPeriod, pivotPeriod)
pivotLow = ta.pivotlow(low, pivotPeriod, pivotPeriod)

// Check if Pivot Points have stayed for the specified candleWait period
var float lastPivotHigh = na
var float lastPivotLow = na
var int barsSincePivotHigh = na
var int barsSincePivotLow = na

if (not na(pivotHigh))
    lastPivotHigh := high[pivotPeriod]
    barsSincePivotHigh := 0
else
    barsSincePivotHigh := na(barsSincePivotHigh) ? na : barsSincePivotHigh + 1

if (not na(pivotLow))
    lastPivotLow := low[pivotPeriod]
    barsSincePivotLow := 0
else
    barsSincePivotLow := na(barsSincePivotLow) ? na : barsSincePivotLow + 1

pivotHighCondition = not na(barsSincePivotHigh) and barsSincePivotHigh == candleWait
pivotLowCondition = not na(barsSincePivotLow) and barsSincePivotLow == candleWait

// Additional filter calculations
rsiValue = ta.rsi(close, rsiLength)
[macdLine, signalLine, _] = ta.macd(close, macdFastLength, macdSlowLength, macdSignalSmoothing)

// Entry conditions based on Pivot Points, EMA, volume, RSI, and MACD
longCondition = (pivotLowCondition or (close > emaValue and volume > avgVolume) or rsiValue < 30 or macdLine > signalLine) and window()
shortCondition = (pivotHighCondition or (close < emaValue and volume > avgVolume) or rsiValue > 70 or macdLine < signalLine) and window()

// Stop-Loss and Trailing Stop settings with manual input
stopLossPercent = input.float(1.0, "Stop-Loss (%)", minval=0.1, maxval=10.0, step=0.1, group="Stop-Loss & Trailing Stop")
trailingStop = input.float(1.5, "Trailing Stop (%)", step=0.1, group="Stop-Loss & Trailing Stop")

// Calculate stop-loss levels
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// Execute trades with reversible signals and regular stop-loss
if (longCondition)
    strategy.entry("Long Entry", strategy.long)
    strategy.exit("Close Long", "Long Entry", stop=longStopLoss, trail_points=trailingStop / 100 * close)

if (shortCondition)
    strategy.entry("Short Entry", strategy.short)
    strategy.exit("Close Short", "Short Entry", stop=shortStopLoss, trail_points=trailingStop / 100 * close)

// Alert Conditions for webhook integration
alertcondition(longCondition, title="Long Entry Signal", message="LONG_ENTRY")
alertcondition(shortCondition, title="Short Entry Signal", message="SHORT_ENTRY")

// Debugging: Plot conditions
plotshape(series=longCondition, title="Long Condition Met", location=location.belowbar, color=color.new(color.green, 0), style=shape.labelup, size=size.small)
plotshape(series=shortCondition, title="Short Condition Met", location=location.abovebar, color=color.new(color.red, 0), style=shape.labeldown, size=size.small)
