//@version=5
strategy('Trend Following Strategy (Reversible) with Trailing Stop-Loss - Improved', shorttitle = 'TrendFollow_Strategy_Rev_TSL_Imp', overlay = true)

// Timing settings
FromMonth = input.int(defval = 4, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(defval = 17, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(defval = 2024, title = 'From Year', minval = 2024)
ToMonth = input.int(defval = 12, title = 'To Month', minval = 1, maxval = 12)  // FIXED: Current month for live
ToDay = input.int(defval = 31, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(defval = 2030, title = 'To Year', minval = 2024)  // FIXED: Near-future for ongoing
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish ? true : false

// Indicator Inputs
fast_length = input.int(10, title = 'Fast MA Length')
slow_length = input.int(30, title = 'Slow MA Length')

// NEW: Filters to Cut Bad Trades
useVolumeFilter = input.bool(true, title="Volume Filter (Vol > 1.5x Avg)?")
useTrendFilter = input.bool(true, title="Trend Strength Filter (ADX > 25)?")
adxLength = input.int(14, title="ADX Length")
minAdx = input.float(25, title="Min ADX for Trade")
signalCooldown = input.int(3, title="Cooldown Bars Between Signals", minval=1)  // NEW: Anti-whipsaw

// Stop-Loss Inputs
trail_offset = input.float(1.5, title = 'Trail Offset %', minval = 0.1, maxval = 5.0, step = 0.1)
breakevenTrigger = input.float(0.5, title="Breakeven % Before Trail", minval=0.1) / 100  // NEW: Guard

// Calculate Moving Averages
fast_ma = ta.sma(close, fast_length)
slow_ma = ta.sma(close, slow_length)

// NEW: Filters
volFilter = not useVolumeFilter or volume > ta.sma(volume, 20) * 1.5
[diplus, diminus, adxValue] = ta.dmi(adxLength, adxLength)  // FIXED: Proper ADX via DMI
trendFilter = not useTrendFilter or adxValue > minAdx

// Plot Moving Averages
plot(fast_ma, color = color.blue, linewidth = 2)
plot(slow_ma, color = color.red, linewidth = 2)

// Strategy Conditions (with filters & cooldown)
var int lastSignalBar = na
cooldownPassed = na(lastSignalBar) or bar_index - lastSignalBar > signalCooldown

longCondition = ta.crossover(fast_ma, slow_ma) and volFilter and trendFilter and window() and cooldownPassed and barstate.isconfirmed
shortCondition = ta.crossunder(fast_ma, slow_ma) and volFilter and trendFilter and window() and cooldownPassed and barstate.isconfirmed

// Track entry for trails
var float longEntry = na
var float shortEntry = na

// Strategy Entry (reversible)
if longCondition
    if strategy.position_size < 0
        strategy.close('Sell')
    strategy.entry('Buy', strategy.long)
    longEntry := close
    lastSignalBar := bar_index

if shortCondition
    if strategy.position_size > 0
        strategy.close('Buy')
    strategy.entry('Sell', strategy.short)
    shortEntry := close
    lastSignalBar := bar_index

// FIXED: Proper Trailing Exits (from entry, after breakeven)
longBreakeven = strategy.position_size > 0 and (close - longEntry) / longEntry >= breakevenTrigger
shortBreakeven = strategy.position_size < 0 and (shortEntry - close) / shortEntry >= breakevenTrigger

if strategy.position_size > 0 and longBreakeven
    trailStopLong = longEntry * (1 + trail_offset / 100)  // FIXED: Trail from entry + %
    strategy.exit('Exit Long', 'Buy', trail_price=trailStopLong, trail_offset=trail_offset / 100 * syminfo.mintick * syminfo.pointvalue)  // FIXED: Proper units

if strategy.position_size < 0 and shortBreakeven
    trailStopShort = shortEntry * (1 - trail_offset / 100)  // FIXED: Trail from entry - %
    strategy.exit('Exit Short', 'Sell', trail_price=trailStopShort, trail_offset=trail_offset / 100 * syminfo.mintick * syminfo.pointvalue)  // FIXED: Proper units

// NEW: Plot filters for debug
plotchar(volFilter and trendFilter ? adxValue : na, title="ADX", color=color.purple, location=location.top)
