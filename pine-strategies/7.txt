//@version=5
strategy("Pivot Points and Volume Color Strategy", overlay=true, max_bars_back=500)

// Inputs for trading settings
length = input.int(50, 'Pivot Length', minval=1, maxval=100)
trailStopPercent = input.float(1.5, 'Trailing Stop (%)', minval=0.1, maxval=5.0)
stopLossPercent = input.float(1.0, 'Stop Loss (%)', minval=0.1, maxval=5.0)

// Timing settings
FromMonth = input.int(4, 'From Month', minval=1, maxval=12)
FromDay = input.int(1, 'From Day', minval=1, maxval=31)
FromYear = input.int(2024, 'From Year', minval=2024)
ToMonth = input.int(12, 'To Month', minval=1, maxval=12)
ToDay = input.int(31, 'To Day', minval=1, maxval=31)
ToYear = input.int(9999, 'To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)

window() =>
    time >= start and time <= finish

volColor() =>
    volume > volume[1] ? true : volume < volume[1] ? false : na()

ph = ta.pivothigh(length, length)
pl = ta.pivotlow(length, length)

// Ensure boolean results for conditions
bool buyCondition = not na(pl) and volColor() != na() and volColor() and window()
bool sellCondition = not na(ph) and volColor() != na() and not volColor() and window()

if (buyCondition)
    if (strategy.position_size <= 0)
        strategy.close_all(comment="Close Sell to Open Buy")
    strategy.entry("Buy Entry", strategy.long)
    strategy.exit("Exit Buy", "Buy Entry", stop=close * (1 - stopLossPercent / 100), trail_price=close * (1 - trailStopPercent / 100), trail_offset=0)

if (sellCondition)
    if (strategy.position_size >= 0)
        strategy.close_all(comment="Close Buy to Open Sell")
    strategy.entry("Sell Entry", strategy.short)
    strategy.exit("Exit Sell", "Sell Entry", stop=close * (1 + stopLossPercent / 100), trail_price=close * (1 + trailStopPercent / 100), trail_offset=0)

plotshape(series=pl and buyCondition, location=location.belowbar, color=color.green, style=shape.triangleup, title="Buy Signal")
plotshape(series=ph and sellCondition, location=location.abovebar, color=color.red, style=shape.triangledown, title="Sell Signal")
