//@version=6
strategy('RSI & Volume Trading Strategy', overlay = true)

// === User Inputs ===
lengthRSI = input.int(14, title = 'RSI Length')
levelUpper = input.float(70.0, title = 'Upper RSI Level')
levelLower = input.float(30.0, title = 'Lower RSI Level')
percentTradeQty = input.float(1.0, title = 'Trade Quantity (%)', minval = 0.1, maxval = 100.0)
percentTrailingStopLoss = input.float(2.0, title = 'Trailing Stop-Loss (%)', minval = 0.1, maxval = 10.0)

// === Timing settings ===
FromMonth = input.int(defval = 1, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(defval = 1, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(defval = 2025, title = 'From Year', minval = 2025)
ToMonth = input.int(defval = 1, title = 'To Month', minval = 1, maxval = 12)
ToDay = input.int(defval = 1, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(defval = 9999, title = 'To Year', minval = 2025)

start  = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() => time >= start and time <= finish

// === RSI Calculation ===
calcRSI = ta.rsi(close, lengthRSI)

// === Volume Conditions ===
increaseVol10 = volume > volume[1] * 1.1
increaseVol30 = volume > volume[1] * 1.3

// === Entry Conditions ===
crossOverLower = ta.crossover(calcRSI, levelLower)
crossUnderUpper = ta.crossunder(calcRSI, levelUpper)
entryLong  = crossOverLower
entryShort = crossUnderUpper

// === Exit Conditions ===
crossUnderUpperExit = ta.crossunder(calcRSI, levelUpper)
crossOverLowerExit  = ta.crossover(calcRSI, levelLower)

// === Trading Entries (with date filter) ===
if window() and entryLong
    strategy.entry('LongEntry', strategy.long, qty = strategy.equity * percentTradeQty / 100 / close, comment = 'Long Entry')
if window() and entryShort
    strategy.entry('ShortEntry', strategy.short, qty = strategy.equity * percentTradeQty / 100 / close, comment = 'Short Entry')

// === Pyramiding Entries (with date filter) ===
if window() and increaseVol10 and entryLong
    strategy.entry('PyramidLong10', strategy.long, qty = strategy.equity * percentTradeQty / 100 / close, comment = 'Pyramid Long 10%')
if window() and increaseVol30 and entryLong
    strategy.entry('PyramidLong30', strategy.long, qty = strategy.equity * percentTradeQty / 100 / close, comment = 'Pyramid Long 30%')

// === Exits (with date filter) ===
if window() and crossUnderUpperExit
    strategy.close('LongEntry', comment = 'Close Long Entry')
if window() and crossOverLowerExit
    strategy.close('ShortEntry', comment = 'Close Short Entry')

// === Trailing Stop-Loss (with date filter) ===
if window()
    strategy.exit('SL_Long', 'LongEntry', trail_price = close * (1 - percentTrailingStopLoss / 100), trail_offset = close * (1 - percentTrailingStopLoss / 100))
if window()
    strategy.exit('SL_Short', 'ShortEntry', trail_price = close * (1 + percentTrailingStopLoss / 100), trail_offset = close * (1 + percentTrailingStopLoss / 100))

// === Plot RSI and Levels ===
plot(calcRSI, 'RSI', color = color.blue)
hline(levelUpper, 'Upper Level', color = color.red)
hline(levelLower, 'Lower Level', color = color.green)
