//@version=6
strategy("Supertrend Strategy with EMA & Volume Filters, Reversal, and Trailing Stop", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15, max_bars_back=4999)

// 1) Time Window Filters
//=============================================================================
useTimeFilter = input.bool(true, title="Enable Time Filter?")
FromMonth     = input.int(1,  title='From Month', minval=1, maxval=12)
FromDay       = input.int(1, title='From Day',   minval=1, maxval=31)
FromYear      = input.int(2025, title='From Year', minval=2024)
ToMonth       = input.int(12,  title='To Month',  minval=1, maxval=12)
ToDay         = input.int(31,  title='To Day',    minval=1, maxval=31)
ToYear        = input.int(9999, title='To Year',  minval=2024)

start  = timestamp(FromYear, FromMonth, FromDay, 0, 0)
finish = timestamp(ToYear,   ToMonth,   ToDay,   23, 59)
timeFilterPass = not useTimeFilter or (time >= start and time <= finish)

// Supertrend inputs
atrPeriod = input(2, "ATR Length")
factor = input.float(0.09, "Factor", step=0.01)
trailingStop = input.float(1.5, "Trailing Stop (%)", step=0.1)
stopLossPercent = input.float(1.0, "Stop-Loss (%)", minval=0.1, maxval=10.0)

// EMA and Volume Filter settings
emaLength = 9
volumeLookback = 50

// EMA and Volume calculation without using lower timeframe data
emaValue = ta.ema(close, emaLength)
avgVolume = ta.sma(volume, volumeLookback)

// Supertrend calculation
[supertrend, direction] = ta.supertrend(factor, atrPeriod)

// Entry conditions based on Supertrend, EMA, and volume
longCondition = (ta.change(direction) < 0) and (close > emaValue) and (volume > avgVolume)
shortCondition = (ta.change(direction) > 0) and (close < emaValue) and (volume > avgVolume)

// Combine conditions for final entry signals
finalLongCondition = longCondition and timeFilterPass
finalShortCondition = shortCondition and timeFilterPass

// Calculate stop-loss levels
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// Execute trades with reversible signals and regular stop-loss
if finalLongCondition
    strategy.close("Short Entry", comment="Close Short on Long Signal") // Close existing short positions
    strategy.entry("Long Entry", strategy.long)
    strategy.exit("Close Long", "Long Entry", stop=longStopLoss, trail_points=trailingStop / 100 * close, trail_offset=trailingStop / 100 * close)

if finalShortCondition
    strategy.close("Long Entry", comment="Close Long on Short Signal") // Close existing long positions
    strategy.entry("Short Entry", strategy.short)
    strategy.exit("Close Short", "Short Entry", stop=shortStopLoss, trail_points=trailingStop / 100 * close, trail_offset=trailingStop / 100 * close)
