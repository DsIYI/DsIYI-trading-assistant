showProbability = input.bool(defval=true, title="Continuation Probability")

showclosingp = input.bool(defval=true, title="Closing Price")
showtable = input.bool(defval=true, title="Temporal Sequence Table", inline="table")
tableoverlay = input.bool(defval=false, title="Main chart", inline="table")
tabletype = input.string(defval="Visual", title="    Table Type", options=["Visual", "Numbers"])
tableposy = input.string(defval='bottom', title='    Location', options=['bottom', 'middle', 'top'], inline='tabpos')
tableposx = input.string(defval='left', title='', options=['left', 'center', 'right'], inline='tabpos')
txtsize = input.string(defval=size.small, title="    Text Size", options=[size.tiny, size.small, size.normal, size.large])
upbarsbg = input.color(defval=#36a845, title="Up Bars", inline="upbc", tooltip="Confirmed, Unconfirmed bars and Border colors")
upbarsbguc = input.color(defval=color.blue, title="", inline="upbc")

dnbarsbg = input.color(defval=#d11d17, title="Down Bars", inline="dnbc", tooltip="Confirmed, Unconfirmed bars and Border colors")

// Trailing Settings
trailing_tp_bars = input.int(10, "Bars to Activate Trailing TP", minval=1, group="Trailing Settings")
trailing_sl_pct = input.float(0.500, "Trailing SL % Offset", minval=0.001, step=0.001, tooltip="Trailing SL offset in %", group="Trailing Settings")

show_trailing_tp = input.bool(true, "Show Trailing TP Level", group="Trailing Settings")
show_trailing_sl = input.bool(true, "Show Trailing SL Level", group="Trailing Settings")
use_fixed_sl = input.bool(false, "Use Fixed SL %", group="Trailing Settings")
fixed_sl_pct = input.float(1.0, "Fixed SL %", minval=0.1, step=0.1, group="Trailing Settings")
use_fixed_tp = input.bool(false, "Use Fixed TP %", group="Trailing Settings")
fixed_tp_pct = input.float(2.0, "Fixed TP %", minval=0.1, step=0.1, group="Trailing Settings")

// Backtest date
start_date = input.time(timestamp("01 Jan 2020 00:00"), "Start Date", group="Backtest Settings")

// Alerts
alertfrequency = input.string(defval=alert.freq_once_per_bar_close, title="Alert Frequency", 
                              options=[alert.freq_once_per_bar_close, alert.freq_once_per_bar], group="Alerts")
longshortalert = input.bool(defval=true, title="Long/Short Alert")
longAmsg = input.string(defval="Trend is Bullish", title=" Long Message", inline="1")
shortAmsg = input.string(defval="Trend is Bearish", title=" Short Message", inline="2")
breakoutalert = input.bool(defval=true, title="Breakout Alert")
bolongAmsg = input.string(defval="Breakout Long", title=" Long Message", inline="3")
boshortAmsg = input.string(defval="Breakout Short", title=" Short Message", inline="4")

// Methods
method addtoarray(float[] this, float value) =>
    this.pop()
    this.unshift(value)
method addtoarray(int[] this, int value) =>
    this.pop()
    this.unshift(value)

getIndex(value) =>
    index = value < 11 ? value - 1 :
             value > 10 and value <= 12 ? 10 :
             value > 12 and value <= 15 ? 11 : 12

getText(value) =>
    txt = value < 11 ? str.tostring(value) :
           value == 11 ? "11-12" :
           value == 12 ? "13-15" : ">15"

method addtrendcount(matrix<int> this, int value, int trend) =>
    index = getIndex(value)
    cc = trend == -1 ? 0 : 1
    this.set(index, cc, 1 + this.get(index, cc))

var trend = close >= open ? 1 : -1
var closes = array.new_float(500, close)
var opens = array.new_float(500, open)
var counts = array.new_int(500, 0)
var trcounts = matrix.new<int>(13, 2, 0)
var count = 0
var trcount = 1
var firstone = true
var breakoutlstyle = breakoutlstylei == "Solid" ? line.style_solid : breakoutlstylei == "Dashed" ? line.style_dashed : line.style_dotted

// Trailing variables
var int entry_bar = na
var float trailing_tp_level = na
var float trailing_sl_level = na
var bool tp_active = false
var int bars_since_entry = na
var float entry_price = na
var float fixed_sl_price = na
var float fixed_tp_price = na
var float last_reverse_price = na

// Precision helper
precise_round(value) => math.round(value * 100000) / 100000

// Core logic
if barstate.isconfirmed and bar_index >= last_bar_index - totalbars and time >= start_date
    if bar_index == last_bar_index - totalbars
        trend := close >= open ? 1 : -1
        count := 0
        trcount := 1
        closes.fill(close)
        opens.fill(open)
    count += 1
    if trend == 1
        if close < math.min(array.min(array.slice(opens, 0, nol)), array.min(array.slice(closes, 0, nol)))
            if firstone
                firstone := false
            else
                trcounts.addtrendcount(trcount, trend)
            trend := -1
            trcount := 0
        if close > closes.get(0) or trend == -1
            opens.addtoarray(trend == -1 ? opens.get(0) : closes.get(0))
            closes.addtoarray(close)
            counts.addtoarray(count)
            count := 0
            trcount += 1
    else if trend == -1
        if close > math.max(array.max(array.slice(opens, 0, nol)), array.max(array.slice(closes, 0, nol)))
            if firstone
                firstone := false
            else
                trcounts.addtrendcount(trcount, trend)
            trend := 1
            trcount := 0
        if close < closes.get(0) or trend == 1
            opens.addtoarray(trend == 1 ? opens.get(0) : closes.get(0))
            closes.addtoarray(close)
            counts.addtoarray(count)
            count := 0
            trcount += 1

// Strategy calculations
trendlevel = trend == 1 ? math.min(array.min(array.slice(opens, 0, nol)), array.min(array.slice(closes, 0, nol))) : 
                          math.max(array.max(array.slice(opens, 0, nol)), array.max(array.slice(closes, 0, nol)))
color trcol = trend != trend[1] ? na : trend == 1 ? uptcol : dntcol
trendlevel_precise = precise_round(trendlevel)
risknotokey = riskmanagement and maximumrisk < math.abs(close - trendlevel_precise) / close and trend != trend[1]
riskokey = trend != trend[1] and (not riskmanagement or maximumrisk >= math.abs(close - trendlevel_precise) / close)

// ========== REVERSE MODE LOGIC ==========
bool reverse_triggered = false
bool reverse_to_long = false
bool reverse_to_short = false

if barstate.isconfirmed and trading_mode == "Reverse on Exit" and strategy.position_size != 0
    if strategy.position_size > 0
        current_sl = use_fixed_sl ? fixed_sl_price : math.min(trendlevel_precise, trailing_sl_level)
        current_tp = use_fixed_tp ? fixed_tp_price : (tp_active ? trailing_tp_level : na)
        
        if not na(current_sl) and (low <= current_sl or close <= current_sl)
            reverse_triggered := true
            reverse_to_short := true
        else if not na(current_tp) and (high >= current_tp or close >= current_tp)
            reverse_triggered := true
            reverse_to_short := true
    
    else if strategy.position_size < 0
        current_sl = use_fixed_sl ? fixed_sl_price : math.max(trendlevel_precise, trailing_sl_level)
        current_tp = use_fixed_tp ? fixed_tp_price : (tp_active ? trailing_tp_level : na)
        
        if not na(current_sl) and (high >= current_sl or close >= current_sl)
            reverse_triggered := true
            reverse_to_long := true
        else if not na(current_tp) and (low <= current_tp or close <= current_tp)
            reverse_triggered := true
            reverse_to_long := true

// Execute reverse
if barstate.isconfirmed and trading_mode == "Reverse on Exit" and reverse_triggered
    strategy.close_all(comment="Reverse Exit")
    
    if reverse_to_long and time >= start_date
        strategy.entry("Long Entry", strategy.long)
        entry_bar := bar_index
        entry_price := close
        last_reverse_price := close
        tp_active := false
        trailing_tp_level := na
        trailing_sl_level := close * (1 - trailing_sl_pct / 100)
        bars_since_entry := 0
        fixed_sl_price := use_fixed_sl ? precise_round(close * (1 - fixed_sl_pct / 100)) : na
        fixed_tp_price := use_fixed_tp ? precise_round(close * (1 + fixed_tp_pct / 100)) : na
        if longshortalert
            alert(longAmsg + " (Reverse)", freq=alertfrequency)
    
    else if reverse_to_short and time >= start_date
        strategy.entry("Short Entry", strategy.short)
        entry_bar := bar_index
        entry_price := close
        last_reverse_price := close
        tp_active := false
        trailing_tp_level := na
        trailing_sl_level := close * (1 + trailing_sl_pct / 100)
        bars_since_entry := 0
        fixed_sl_price := use_fixed_sl ? precise_round(close * (1 + fixed_sl_pct / 100)) : na
        fixed_tp_price := use_fixed_tp ? precise_round(close * (1 - fixed_tp_pct / 100)) : na
        if longshortalert
            alert(shortAmsg + " (Reverse)", freq=alertfrequency)

// ========== CLASSIC MODE ENTRIES ==========
if barstate.isconfirmed and trading_mode == "Classic" and showTrend and trend == 1 and riskokey and strategy.position_size == 0 and time >= start_date
    strategy.entry("Long Entry", strategy.long)
    entry_bar := bar_index
    entry_price := close
    tp_active := false
    trailing_tp_level := na
    trailing_sl_level := close * (1 - trailing_sl_pct / 100)
    bars_since_entry := 0
    fixed_sl_price := use_fixed_sl ? precise_round(close * (1 - fixed_sl_pct / 100)) : na
    fixed_tp_price := use_fixed_tp ? precise_round(close * (1 + fixed_tp_pct / 100)) : na

if barstate.isconfirmed and trading_mode == "Classic" and showTrend and trend == -1 and riskokey and strategy.position_size == 0 and time >= start_date
    strategy.entry("Short Entry", strategy.short)
    entry_bar := bar_index
    entry_price := close
    tp_active := false
    trailing_tp_level := na
    trailing_sl_level := close * (1 + trailing_sl_pct / 100)
    bars_since_entry := 0
    fixed_sl_price := use_fixed_sl ? precise_round(close * (1 + fixed_sl_pct / 100)) : na
    fixed_tp_price := use_fixed_tp ? precise_round(close * (1 - fixed_tp_pct / 100)) : na

// ========== INITIAL ENTRY (Reverse mode) ==========
if barstate.isconfirmed and trading_mode == "Reverse on Exit" and strategy.opentrades == 0 and strategy.closedtrades == 0 and showTrend and time >= start_date
    if trend == 1 and riskokey
        strategy.entry("Long Entry", strategy.long)
        entry_bar := bar_index
        entry_price := close
        tp_active := false
        trailing_tp_level := na
        trailing_sl_level := close * (1 - trailing_sl_pct / 100)
        bars_since_entry := 0
        fixed_sl_price := use_fixed_sl ? precise_round(close * (1 - fixed_sl_pct / 100)) : na
        fixed_tp_price := use_fixed_tp ? precise_round(close * (1 + fixed_tp_pct / 100)) : na
    else if trend == -1 and riskokey
        strategy.entry("Short Entry", strategy.short)
        entry_bar := bar_index
        entry_price := close
        tp_active := false
        trailing_tp_level := na
        trailing_sl_level := close * (1 + trailing_sl_pct / 100)
        bars_since_entry := 0
        fixed_sl_price := use_fixed_sl ? precise_round(close * (1 + fixed_sl_pct / 100)) : na
        fixed_tp_price := use_fixed_tp ? precise_round(close * (1 - fixed_tp_pct / 100)) : na

// Track bars
if strategy.position_size != 0 and barstate.isconfirmed
    bars_since_entry := bar_index - entry_bar

// Update Trailing SL
if barstate.isconfirmed and strategy.position_size > 0
    new_sl = precise_round(close * (1 - trailing_sl_pct / 100))
    trailing_sl_level := math.max(na(trailing_sl_level) ? new_sl : trailing_sl_level, new_sl)

if barstate.isconfirmed and strategy.position_size < 0
    new_sl = precise_round(close * (1 + trailing_sl_pct / 100))
    trailing_sl_level := math.min(na(trailing_sl_level) ? new_sl : trailing_sl_level, new_sl)

// Update Trailing TP
if barstate.isconfirmed and strategy.position_size > 0
    if not na(bars_since_entry) and bars_since_entry >= trailing_tp_bars
        tp_active := true
        new_tp = precise_round(close * (1 + trailing_tp_pct / 100))
        trailing_tp_level := math.max(na(trailing_tp_level) ? new_tp : trailing_tp_level, new_tp)

if barstate.isconfirmed and strategy.position_size < 0
    if not na(bars_since_entry) and bars_since_entry >= trailing_tp_bars
        tp_active := true
        new_tp = precise_round(close * (1 - trailing_tp_pct / 100))
        trailing_tp_level := math.min(na(trailing_tp_level) ? new_tp : trailing_tp_level, new_tp)

// Exits (Classic mode only)
if barstate.isconfirmed and trading_mode == "Classic" and strategy.position_size > 0
    sl_price = use_fixed_sl ? fixed_sl_price : math.min(trendlevel_precise, trailing_sl_level)
    tp_price = use_fixed_tp ? fixed_tp_price : (tp_active ? trailing_tp_level : na)
    strategy.exit("Long Exit", "Long Entry", stop=sl_price, limit=tp_price)

if barstate.isconfirmed and trading_mode == "Classic" and strategy.position_size < 0
    sl_price = use_fixed_sl ? fixed_sl_price : math.max(trendlevel_precise, trailing_sl_level)
    tp_price = use_fixed_tp ? fixed_tp_price : (tp_active ? trailing_tp_level : na)
    strategy.exit("Short Exit", "Short Entry", stop=sl_price, limit=tp_price)

// ========== PLOTS ==========
plot(showTrend and bar_index >= last_bar_index - totalbars ? trendlevel : na, color=trcol, linewidth=trendwidth, title="Trend Level")
plot(show_trailing_tp and tp_active ? trailing_tp_level : na, color=color.yellow, linewidth=1, title="Trailing TP")
plot(show_trailing_sl and not na(trailing_sl_level) and strategy.position_size != 0 ? trailing_sl_level : na, color=color.orange, linewidth=1, title="Trailing SL")
plotshape(showTrend and risknotokey ? trendlevel : na, location=location.absolute, style=shape.xcross, size=size.small, color=color.gray)

// Classic mode markers
plotshape(trading_mode == "Classic" and showTrend and trend == 1 and riskokey ? trendlevel : na, 
          text="Long", color=color.lime, textcolor=color.black, location=location.absolute, style=shape.labelup)
plotshape(trading_mode == "Classic" and showTrend and trend == -1 and riskokey ? trendlevel : na, 
          text="Short", color=color.red, textcolor=color.black, location=location.absolute, style=shape.labeldown)

// Reverse mode markers
plotshape(trading_mode == "Reverse on Exit" and reverse_to_long ? close : na, 
          text="⚡LONG", color=color.new(color.blue, 0), textcolor=color.white, 
          location=location.absolute, style=shape.labelup, size=size.normal)
plotshape(trading_mode == "Reverse on Exit" and reverse_to_short ? close : na, 
          text="⚡SHORT", color=color.new(color.maroon, 0), textcolor=color.white, 
          location=location.absolute, style=shape.labeldown, size=size.normal)

// Entry price line extending to next bars
var line entry_line = na
if reverse_triggered and not na(last_reverse_price)
    line.delete(entry_line)
    entry_line := line.new(bar_index, last_reverse_price, bar_index + 5, last_reverse_price, 
                           color=reverse_to_long ? color.blue : color.maroon, 
                           width=2, style=line.style_dashed)

bgcolor(reverse_triggered ? color.new(color.yellow, 85) : na, title="Reverse Bar")

// Boxes drawing
var array<box> all_boxes = array.new<box>()
if barstate.isconfirmed and matrix.max(trcounts) > 0
    while array.size(all_boxes) > 500
        box.delete(array.shift(all_boxes))
    
    array<int> ccounts = array.copy(counts)
    ccounts.unshift(count + 1)
    
    int xloc = bar_index
    for x = 0 to math.min(closes.size() - 1, 50)
        if x < closes.size() - 1
            if bar_index - x < 1 or (closes.get(x) == closes.get(x + 1) and opens.get(x) == opens.get(x + 1))
                continue
        
        bgcol = closes.get(x) > opens.get(x) ? upbarsbg : dnbarsbg
        bdcol = closes.get(x) > opens.get(x) ? upbarsbd : dnbarsbd
        if flipflop and x > 0 and x < closes.size() - 1
            if (closes.get(x) > opens.get(x) and closes.get(x - 1) < opens.get(x - 1) and closes.get(x + 1) < opens.get(x + 1)) or
               (closes.get(x) < opens.get(x) and closes.get(x - 1) > opens.get(x - 1) and closes.get(x + 1) > opens.get(x + 1))
                bgcol := flipflopcol
                
        left = gap == "On" ? xloc - ccounts.get(x) : bar_index - x - 1
        right = gap == "On" ? xloc : bar_index - x
        newbox = box.new(left=left, top=closes.get(x), right=right, bottom=opens.get(x), border_width=1, border_color=bdcol, bgcolor=bgcol)
        array.push(all_boxes, newbox)
        xloc -= ccounts.get(x)

// Breakout lines
var array<line> all_lines = array.new<line>()
if barstate.isconfirmed and breakout and closes.size() > 12 and matrix.max(trcounts) > 0
    while array.size(all_lines) > 100
        line.delete(array.shift(all_lines))
    
    array<int> ccounts = array.copy(counts)
    ccounts.unshift(count + 1)
    
    int xloc_break = bar_index
    for x = 0 to math.min(closes.size() - 10, 30)
        t = closes.get(x) > opens.get(x) ? 1 : -1
        bout = 0.
        minv = opens.get(x)
        maxv = opens.get(x)
        int xloc1 = bar_index
        for y = 0 to closes.size() - 2
            if y - x > 150
                break
            if y < x + 3
                if y > x and (t == 1 and closes.get(y) > closes.get(x) or t == -1 and closes.get(y) < closes.get(x))
                    break
                xloc1 -= ccounts.get(y)
                if y > x
                    minv := math.min(minv, closes.get(y))
                    maxv := math.max(maxv, closes.get(y))
                continue
            
            if (t == 1 and closes.get(x) < closes.get(y)) or (t == -1 and closes.get(x) > closes.get(y))
                break
            if t == 1 and maxv <= closes.get(y) and closes.get(x) > closes.get(y) and closes.get(y) > opens.get(y) and closes.get(x + 1) <= closes.get(y) and closes.get(y) > math.max(closes.get(y - 1), closes.get(y + 1)) 
                bout := closes.get(y) - closes.get(x)
            else if t == -1 and minv >= closes.get(y) and closes.get(x) < closes.get(y) and closes.get(y) < opens.get(y) and closes.get(x + 1) >= closes.get(y) and closes.get(y) < math.min(closes.get(y - 1), closes.get(y + 1))
                bout := closes.get(y) - closes.get(x)
            if bout != 0
                left = gap == "On" ? xloc1 - ccounts.get(y) : bar_index - y - 1
                right = gap == "On" ? xloc_break : bar_index - x
                col = t == 1 ? breakoutucol : breakoutdcol
                if breakoutalert and x == 0
                    if t == 1
                        alert(bolongAmsg, freq=alertfrequency)
                    else
                        alert(boshortAmsg, freq=alertfrequency)
                newline = line.new(x1=left + 1, y1=closes.get(x) + bout, x2=right, y2=closes.get(x) + bout, color=col, style=breakoutlstyle)
                array.push(all_lines, newline)
                break
            xloc1 -= ccounts.get(y)
        xloc_break -= ccounts.get(x)

// Alerts (Classic mode only)
if barstate.isconfirmed and trading_mode == "Classic" and longshortalert
    if trend == 1 and riskokey
        alert(longAmsg, freq=alertfrequency)
    else if trend == -1 and riskokey
        alert(shortAmsg, freq=alertfrequency)

// Table
if barstate.isconfirmed and showtable and matrix.max(trcounts) > 0
    var tsectable1 = table.new(position=tableposy + '_' + tableposx, columns=22, rows=20, frame_width=1, bgcolor=chart.bg_color, frame_color=chart.fg_color, border_width=1, border_color=color.new(chart.fg_color, 70), force_overlay=true)
    var tsectable2 = table.new(position=tableposy + '_' + tableposx, columns=22, rows=20, frame_width=1, bgcolor=chart.bg_color, frame_color=chart.fg_color, border_width=1, border_color=color.new(chart.fg_color, 70))
   
    tsectable = tableoverlay ? tsectable1 : tsectable2
    cIndex = getIndex(trcount)
    
    table.clear(tsectable, 0, 0, 21, 19)
    
    for y = 2 to 14
        table.cell(tsectable, 0, y, text=getText(y - 1), text_color=chart.bg_color, bgcolor=chart.fg_color, text_size=txtsize)
  
    table.cell(tsectable, 0, cIndex + 2, text=getText(cIndex + 1), text_color=color.white, bgcolor=color.blue, text_size=txtsize)

    var trendrate = array.from(1, 1, 1, 14, 35, 54, 58, 63, 70, 75, 80, 90, 100)
    utotal = 0.
    dtotal = 0.
    usum = 0.
    dsum = 0.
    for y = 12 to 0
        dtotal += (trcounts.get(y, 0) * trendrate.get(y)) * (y + 1)
        utotal += (trcounts.get(y, 1) * trendrate.get(y)) * (y + 1)
        dsum += trcounts.get(y, 0) * (y + 1)
        usum += trcounts.get(y, 1) * (y + 1)

    utrending = utotal / usum
    dtrending = dtotal / dsum
    uptrending = "Trending: " + str.tostring(utrending, '#.#') + "% \nChoppy: " + str.tostring(100 - utrending, '#.#')
    dntrending = "Trending: " + str.tostring(dtrending, '#.#') + "% \nChoppy: " + str.tostring(100 - dtrending, '#.#')

    maxcol = tabletype == "Visual" ? math.min(matrix.max(trcounts), 10) : 1
    float step = matrix.max(trcounts) / maxcol
