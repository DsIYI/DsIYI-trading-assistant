//@version=5
strategy("Pivot Point Strategy with EMA & Volume Filters, Reversal, and Trailing Stop", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15, max_bars_back=4999)

// Timing settings
FromMonth = input.int(defval=4, title='From Month', minval=1, maxval=12)
FromDay = input.int(defval=17, title='From Day', minval=1, maxval=31)
FromYear = input.int(defval=2024, title='From Year', minval=2024)
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12)
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31)
ToYear = input.int(defval=9999, title='To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish

// Pivot Point settings
pivotPeriod = input.int(defval=5, title="Pivot Period", minval=1, maxval=50)
candleWait = input.int(defval=1, title="Candle Wait Period", options=[1, 2, 3, 5])

// EMA and Volume Filter settings
emaLength = 9
volumeLookback = 50

// NEW: Cooldown to prevent rapid flips
signalCooldownBars = input.int(3, title="Signal Cooldown Bars (Anti-Whipsaw)", minval=1, maxval=10)
var int lastSignalBar = na

// EMA and Volume calculation
emaValue = ta.ema(close, emaLength)  // FIXED: Use emaLength, not emaValue
avgVolume = ta.sma(volume, volumeLookback)

// Pivot Point calculations
pivotHigh = ta.pivothigh(high, pivotPeriod, pivotPeriod)
pivotLow = ta.pivotlow(low, pivotPeriod, pivotPeriod)

// Plot Pivot Points
plotshape(series=pivotHigh, location=location.abovebar, color=color.red, style=shape.labeldown, title="Pivot High")
plotshape(series=pivotLow, location=location.belowbar, color=color.green, style=shape.labelup, title="Pivot Low")

// Check if Pivot Points have stayed for the specified candleWait period
var float lastPivotHigh = na
var float lastPivotLow = na
var int barsSincePivotHigh = na
var int barsSincePivotLow = na

if (not na(pivotHigh))
    lastPivotHigh := high[pivotPeriod]
    barsSincePivotHigh := 0
else
    barsSincePivotHigh := na(barsSincePivotHigh) ? na : barsSincePivotHigh + 1

if (not na(pivotLow))
    lastPivotLow := low[pivotPeriod]
    barsSincePivotLow := 0
else
    barsSincePivotLow := na(barsSincePivotLow) ? na : barsSincePivotLow + 1

pivotHighCondition = not na(barsSincePivotHigh) and barsSincePivotHigh == candleWait
pivotLowCondition = not na(barsSincePivotLow) and barsSincePivotLow == candleWait

// Combined Entry conditions (Pivot + EMA/Vol, with window & confirmed)
longCondition = (pivotLowCondition or (close > emaValue and volume > avgVolume)) and window() and barstate.isconfirmed and (na(lastSignalBar) or bar_index - lastSignalBar > signalCooldownBars)
shortCondition = (pivotHighCondition or (close < emaValue and volume > avgVolume)) and window() and barstate.isconfirmed and (na(lastSignalBar) or bar_index - lastSignalBar > signalCooldownBars)

// Stop-Loss and Trailing Stop settings
stopLossPercent = input.float(1.0, "Stop-Loss (%)", minval=0.001, maxval=10.0)
trailingPercent = input.float(1.5, "Trailing Stop (%)", step=0.01) / 100  // FIXED: Proper % trail

// Track entry for trails
var float longEntry = na
var float shortEntry = na

// Execute trades with reversible signals, position gates, and fixed trails
if (longCondition)
    if strategy.position_size < 0  // FIXED: Only close if in short
        strategy.close("Short Entry", comment="Close Short on Long Signal")
    strategy.entry("Long Entry", strategy.long)
    longEntry := close  // FIXED: Hold entry for trail
    lastSignalBar := bar_index

if (shortCondition)
    if strategy.position_size > 0  // FIXED: Only close if in long
        strategy.close("Long Entry", comment="Close Long on Short Signal")
    strategy.entry("Short Entry", strategy.short)
    shortEntry := close  // FIXED: Hold entry for trail
    lastSignalBar := bar_index

// FIXED: Proper trailing exits (from entry, activates after breakeven)
if strategy.position_size > 0
    trailStopLong = longEntry * (1 + trailingPercent)  // Trail up from entry + %
    strategy.exit("Trail Long", "Long Entry", trail_price=trailStopLong, trail_offset=trailingPercent * longEntry)  // FIXED: Proper params

if strategy.position_size < 0
    trailStopShort = shortEntry * (1 - trailingPercent)  // Trail down from entry - %
    strategy.exit("Trail Short", "Short Entry", trail_price=trailStopShort, trail_offset=trailingPercent * shortEntry)  // FIXED: Proper params

// Initial SL (fixed % from entry)
if strategy.position_size > 0 and na(longEntry[1])
    strategy.exit("SL Long", "Long Entry", stop=longEntry * (1 - stopLossPercent / 100))
if strategy.position_size < 0 and na(shortEntry[1])
    strategy.exit("SL Short", "Short Entry", stop=shortEntry * (1 + stopLossPercent / 100))

// Debugging: Plot conditions
plotshape(series=longCondition, title="Long Condition", location=location.belowbar, color=color.green, style=shape.labelup, size=size.small)
plotshape(series=shortCondition, title="Short Condition", location=location.abovebar, color=color.red, style=shape.labeldown, size=size.small)
