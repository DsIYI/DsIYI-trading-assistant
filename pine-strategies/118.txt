//@version=6
strategy("Engulfing Strategy with Dynamic Trailing Stop", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15)

// === INPUTS ===
// Time Window Inputs
useTimeFilter = input.bool(true, title="Enable Time Filter?")
FromMonth = input.int(1, title="From Month", minval=1, maxval=12)
FromDay = input.int(1, title="From Day", minval=1, maxval=31)
FromYear = input.int(2017, title="From Year", minval=2017)
ToMonth = input.int(12, title="To Month", minval=1, maxval=12)
ToDay = input.int(31, title="To Day", minval=1, maxval=31)
ToYear = input.int(9999, title="To Year", minval=2017)

// Trailing Stop-Loss and Take-Profit Inputs
trailingStopPercent = input.float(1.0, title="Trailing Stop-Loss (%)", minval=0.001, maxval=10.0, step=0.001)
trailingTakeProfitPercent = input.float(1.0, title="Trailing Take-Profit (%)", minval=0.001, maxval=10.0, step=0.001)

// === TIME FILTER ===
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
timeFilterPass = not useTimeFilter or (time >= start and time <= finish)

// === ENGULFING CANDLES LOGIC ===
openBarPrevious = open[1]
closeBarPrevious = close[1]
openBarCurrent = open
closeBarCurrent = close

bullishEngulfing = (openBarCurrent <= closeBarPrevious) and (openBarCurrent < openBarPrevious) and (closeBarCurrent > openBarPrevious)
bearishEngulfing = (openBarCurrent >= closeBarPrevious) and (openBarCurrent > openBarPrevious) and (closeBarCurrent < openBarPrevious)

// Plot signals
plotshape(bullishEngulfing, style=shape.triangleup, location=location.belowbar, color=color.green, size=size.tiny)
plotshape(bearishEngulfing, style=shape.triangledown, location=location.abovebar, color=color.red, size=size.tiny)

// === ENTRY CONDITIONS ===
finalLongCondition = bullishEngulfing and timeFilterPass
finalShortCondition = bearishEngulfing and timeFilterPass

// === DYNAMIC TRAILING LOGIC ===
// Variables to track state
var float longEntryPrice = 0
var float shortEntryPrice = 0
var bool longInProfit = false
var bool shortInProfit = false
var float longBreakEven = 0
var float shortBreakEven = 0

// Update entry prices and break-even levels
if finalLongCondition
    longEntryPrice := close
    longBreakEven := close * (1 + trailingStopPercent / 100)
    longInProfit := false

if finalShortCondition
    shortEntryPrice := close
    shortBreakEven := close * (1 - trailingStopPercent / 100)
    shortInProfit := false

// Check if price reaches break-even to switch to trailing take-profit
if strategy.position_size > 0 and close >= longBreakEven
    longInProfit := true
if strategy.position_size < 0 and close <= shortBreakEven
    shortInProfit := true

// Reset to trailing stop-loss if price reverses before take-profit
if strategy.position_size > 0 and longInProfit and close < longBreakEven
    longInProfit := false
if strategy.position_size < 0 and shortInProfit and close > shortBreakEven
    shortInProfit := false

// === TRADE EXECUTION ===
// Long Trades
if finalLongCondition
    strategy.close("Short Entry", comment="Close Short on Long Signal")
    strategy.entry("Long Entry", strategy.long)
    strategy.exit("Close Long", "Long Entry", trail_points=longInProfit ? trailingTakeProfitPercent / 100 * close : trailingStopPercent / 100 * close, trail_offset=longInProfit ? trailingTakeProfitPercent / 100 * close : trailingStopPercent / 100 * close)

// Short Trades
if finalShortCondition
    strategy.close("Long Entry", comment="Close Long on Short Signal")
    strategy.entry("Short Entry", strategy.short)
    strategy.exit("Close Short", "Short Entry", trail_points=shortInProfit ? trailingTakeProfitPercent / 100 * close : trailingStopPercent / 100 * close, trail_offset=shortInProfit ? trailingTakeProfitPercent / 100 * close : trailingStopPercent / 100 * close)

// === ALERTS ===
alertcondition(bullishEngulfing, title="Bullish Engulfing", message="[CurrencyPair] [TimeFrame], Bullish candle engulfing previous candle")
alertcondition(bearishEngulfing, title="Bearish Engulfing", message="[CurrencyPair] [TimeFrame], Bearish candle engulfing previous candle")
