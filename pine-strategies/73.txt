//@version=5
indicator(title='Extended Non-Reversal Indicator w/ Alerts (Real-Time Execution)', overlay=true)

// Timing Inputs
FromMonth = input.int(defval=7, title='From Month', minval=1, maxval=12, group='Timing')
FromDay   = input.int(defval=5, title='From Day', minval=1, maxval=31, group='Timing')
FromYear  = input.int(defval=2024, title='From Year', minval=2024, group='Timing')
ToMonth   = input.int(defval=1, title='To Month', minval=1, maxval=12, group='Timing')
ToDay     = input.int(defval=1, title='To Day', minval=1, maxval=31, group='Timing')
ToYear    = input.int(defval=9999, title='To Year', minval=2024, group='Timing')
start     = timestamp(FromYear, FromMonth, FromDay, 0, 0)
finish    = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window()  => time >= start and time <= finish

// Pivot Points
pivotPeriod = input.int(defval=5, title='Pivot Period', minval=1, maxval=50, group='Pivot Points')
candleWait  = input.int(defval=1, title='Candle Wait Period', options=[1, 2, 3, 5], group='Pivot Points')

// EMA & Volume Filter
emaLength      = input.int(defval=9, title='EMA Length', minval=1, maxval=100, group='EMA & Volume Filter')
volumeLookback = input.int(defval=50, title='Volume Lookback Period', minval=1, maxval=100, group='EMA & Volume Filter')

// RSI Filter
rsiLength = input.int(defval=14, title='RSI Length', minval=1, maxval=50, group='RSI Filter')

// MACD Filter
macdFastLength      = input.int(defval=12, title='MACD Fast Length', minval=1, maxval=50, group='MACD Filter')
macdSlowLength      = input.int(defval=26, title='MACD Slow Length', minval=1, maxval=50, group='MACD Filter')
macdSignalSmoothing = input.int(defval=9, title='MACD Signal Smoothing', minval=1, maxval=50, group='MACD Filter')

// Stop-Loss & Trailing Stop
stopLossPercent = input.float(defval=1.0, title='Stop-Loss (%)', minval=0.1, maxval=10.0, step=0.1, group='Stop-Loss & Trailing Stop')

// Trade Direction Control
tradeDirection = input.string(defval='Both', title='Trade Direction Allowed', options=['Buy', 'Sell', 'Both'], group='Trade Mode')

// Entry Conditions
emaValue = ta.ema(close, emaLength)
rsiValue = ta.rsi(close, rsiLength)
[macdLine, signalLine, _] = ta.macd(close, macdFastLength, macdSlowLength, macdSignalSmoothing)

longConditionEMA = (close > emaValue) and (volume > ta.sma(volume, volumeLookback)) and window()
shortConditionEMA = (close < emaValue) and (volume > ta.sma(volume, volumeLookback)) and window()
longConditionRSI = (rsiValue < 30) and window()
shortConditionRSI = (rsiValue > 70) and window()
longConditionMACD = (macdLine > signalLine) and window()
shortConditionMACD = (macdLine < signalLine) and window()

longCondition = longConditionEMA or longConditionRSI or longConditionMACD
shortCondition = shortConditionEMA or shortConditionRSI or shortConditionMACD

// Stop-Loss Calculation
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// Plot Buy and Sell Signals
plotshape(series=longCondition, location=location.belowbar, color=color.green, style=shape.labelup, title='Buy Signal')
plotshape(series=shortCondition, location=location.abovebar, color=color.red, style=shape.labeldown, title='Sell Signal')

// Alerts for Immediate Execution (No Bar Close Delay)
alertcondition(longCondition, title='BUY ALERT', message='{"symbol": "SOLUSDT", "side": "buy", "type": "market", "quantity": 1, "leverage": 10}')
alertcondition(shortCondition, title='SELL ALERT', message='{"symbol": "SOLUSDT", "side": "sell", "type": "market", "quantity": 1, "leverage": 10}')
alertcondition(longCondition and close <= longStopLoss, title='STOP-LOSS LONG', message='{"symbol": "SOLUSDT", "side": "sell", "type": "market", "quantity": 1}')
alertcondition(shortCondition and close >= shortStopLoss, title='STOP-LOSS SHORT', message='{"symbol": "SOLUSDT", "side": "buy", "type": "market", "quantity": 1}')
