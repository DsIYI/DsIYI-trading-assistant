//@version=5
strategy("AI-2-1-Second Scalper • Multi-Asset • SOL/MYX Auto Optimized • V5.0",
     overlay = true,
     max_bars_back = 5000,
     initial_capital = 100,  // ← You're risking from $100 account
     default_qty_type = strategy.fixed,
     default_qty_value = 0,
     commission_type = strategy.commission.percent,
     commission_value = 0.06)

// ───────────────────────────── USER INPUTS ────────────────────────────
use1s        = input.bool(true,   "Allow 1-second charts (requires Premium)")
secTF_input  = input.string("1S", "1s TF string (for info only)")
assetProfile = input.string("SOLUSDT.P", "Asset Profile", options=["SOLUSDT.P", "MYXUSDT.P"])

// Auto-configure based on asset
leverage     = assetProfile == "SOLUSDT.P" ? 100 : 25
maxRiskPercent = assetProfile == "SOLUSDT.P" ? 1.0 : 3.0  // % of $100 → $1 or $3

// Qty Rules (Bybit API)
minLot       = assetProfile == "SOLUSDT.P" ? 0.1  : 0.001  // SOL: 0.1, MYX: check actuals
qtyDecimals  = assetProfile == "SOLUSDT.P" ? 1    : 3      // SOL: step 0.1, MYX: often 0.001
tickSize     = assetProfile == "SOLUSDT.P" ? 0.01 : 0.001  // SOL: $0.01, MYX: often $0.001
minNotional  = 5.0  // Bybit minimum for both

// EMA & Pivot Settings (optimized per asset)
fastEMA      = assetProfile == "SOLUSDT.P" ? 6  : 8
slowEMA      = assetProfile == "SOLUSDT.P" ? 15 : 21
leftBars     = assetProfile == "SOLUSDT.P" ? 3  : 4
rightBars    = assetProfile == "SOLUSDT.P" ? 3  : 4

// Stop Loss / Take Profit (adjusted for volatility)
slATRmult    = assetProfile == "SOLUSDT.P" ? 0.8 : 1.0
tpATRmult    = assetProfile == "SOLUSDT.P" ? 1.5 : 1.2

// Trailing & Hard Stops
trailingTP_pct = assetProfile == "SOLUSDT.P" ? 0.3 : 0.5
hardStop_pct   = assetProfile == "SOLUSDT.P" ? 1.5 : 3.0  // Match risk %
moveToBE       = true
tp1Mult        = 0.5

// Strategy Mode
strategyMode = input.string("Normal", "Strategy mode", options = ["Normal", "Reverse", "TP-only"])

// Filters
useVolumeFilter = true
useRSIFilter    = true
useADX          = assetProfile == "SOLUSDT.P" ? true : false  // SOL trends stronger

// HTF Filter
useHTF       = true
htfTF        = assetProfile == "SOLUSDT.P" ? "3" : "1"  // SOL: smoother, MYX: faster
htfEMAlen    = 100

// Backtest Period
startDate    = input.time(timestamp("1 May 2025 00:00 +0000"), "Back-test FROM")
endDate      = input.time(timestamp("1 Jan 2099 00:00 +0000"), "Back-test TO")
inTradePeriod = (time >= startDate) and (time <= endDate)

// Execution Timing
onBarClose   = input.bool(false, "Require 'on bar close' for entries?")

// Drawdown Protection (of total equity)
maxDrawdownPct = input.float(20.0, "Max Drawdown % (halt trading)", step=1.0)

// Slippage
slippageTicks = assetProfile == "SOLUSDT.P" ? 1 : 2  // MYX may need more slippage

// ATR Settings
atrLen       = 14
atrRiskMult  = 1.0

// ───────────────────────────── CORE INDICATORS ───────────────────────
fEMA = ta.ema(close, fastEMA)
sEMA = ta.ema(close, slowEMA)

// HTF — Sampled only at bar close
htfClose = request.security(syminfo.tickerid, htfTF, close, lookahead = barmerge.lookahead_off)
htfEMA   = ta.ema(htfClose, htfEMAlen)

var bool htfTrendUp = false
var bool htfTrendDn = false
if barstate.isconfirmed
    htfTrendUp := htfClose > htfEMA
    htfTrendDn := htfClose < htfEMA
else
    htfTrendUp := htfTrendUp[1]
    htfTrendDn := htfTrendDn[1]

// Pivots — Sampled only at bar close
pivotH = ta.pivothigh(high, leftBars, rightBars)
pivotL = ta.pivotlow(low, leftBars, rightBars)

var bool validPivotL = false
var bool validPivotH = false
if barstate.isconfirmed
    validPivotL := not na(pivotL)
    validPivotH := not na(pivotH)
else
    validPivotL := validPivotL[1]
    validPivotH := validPivotH[1]

// Volatility & Filters — Sampled only at bar close
atr = ta.atr(atrLen)
rsi = ta.rsi(close, 14)
volAvg = ta.sma(volume, 20)
[dmiPlus, dmiMinus, adx] = ta.dmi(14, 14)

var bool volumeFilterConfirmed = false
var bool rsiLongFilterConfirmed = false
var bool rsiShortFilterConfirmed = false
var bool adxFilterConfirmed = false

if barstate.isconfirmed
    volumeFilterConfirmed := not useVolumeFilter or (volume > volAvg * 1.5)
    rsiLongFilterConfirmed := not useRSIFilter or (rsi < 70)
    rsiShortFilterConfirmed := not useRSIFilter or (rsi > 30)
    adxFilterConfirmed := not useADX or (adx > 20)
else
    volumeFilterConfirmed := volumeFilterConfirmed[1]
    rsiLongFilterConfirmed := rsiLongFilterConfirmed[1]
    rsiShortFilterConfirmed := rsiShortFilterConfirmed[1]
    adxFilterConfirmed := adxFilterConfirmed[1]

// Bias
bullBias = not useHTF or htfTrendUp
bearBias = not useHTF or htfTrendDn

// Base Signals
longSignalBase  = validPivotL and (fEMA > sEMA)
shortSignalBase = validPivotH and (fEMA < sEMA)

// Apply Filters
longSignal  = longSignalBase and bullBias and volumeFilterConfirmed and rsiLongFilterConfirmed and adxFilterConfirmed
shortSignal = shortSignalBase and bearBias and volumeFilterConfirmed and rsiShortFilterConfirmed and adxFilterConfirmed

// Strategy Mode
if strategyMode == "Reverse"
    tmp = longSignal
    longSignal  := shortSignal
    shortSignal := tmp
if strategyMode == "TP-only"
    longSignal := false
    shortSignal := false

// Position Check
noPosition = strategy.position_size == 0

// Drawdown Protection
peakEquity = ta.highest(strategy.equity, 500)
drawdownPct = (peakEquity - strategy.equity) / peakEquity * 100
tooMuchDrawdown = drawdownPct > maxDrawdownPct

// === NON-REPAINTING FINAL SIGNALS ===
var bool finalLongConfirmed  = false
var bool finalShortConfirmed = false

if barstate.isconfirmed
    entryCondition = not onBarClose or barstate.isconfirmed
    finalLongConfirmed  := inTradePeriod and longSignal and entryCondition and noPosition and not tooMuchDrawdown
    finalShortConfirmed := inTradePeriod and shortSignal and entryCondition and noPosition and not tooMuchDrawdown
else
    finalLongConfirmed  := finalLongConfirmed[1]
    finalShortConfirmed := finalShortConfirmed[1]

// ───────────────────────────── POSITION SIZING ───────────────────────
price = close
stopDistanceUSD = slATRmult * atr
stopDistanceUSD := math.max(stopDistanceUSD, tickSize)  // Use asset-specific tick

dollarRisk = strategy.equity * (maxRiskPercent / 100.0)
rawQty = stopDistanceUSD > 0 ? (dollarRisk / stopDistanceUSD) * leverage : (5.0 / price) * leverage  // Fallback to min notional

// Round to correct decimal places
qty = math.round(rawQty * math.pow(10, qtyDecimals)) / math.pow(10, qtyDecimals)
qty := math.max(qty, minLot)

// Enforce Bybit min notional
notionalValue = qty * price
if notionalValue < minNotional
    qty := 0  // Cancel trade if below exchange minimum

validQty = qty >= minLot and notionalValue >= minNotional

// ───────────────────────────── EXIT SETUP ────────────────────────────
mintick = tickSize  // Override syminfo.mintick with asset-specific tick

profitTicks = math.max(1, math.round((tpATRmult * atr) / mintick))
lossTicks   = math.max(1, math.round((slATRmult * atr) / mintick))

trailingTicks = trailingTP_pct > 0.0 ? math.max(1, math.round(((trailingTP_pct/100.0) * price) / mintick)) : na
hardStopTicks = hardStop_pct   > 0.0 ? math.max(1, math.round(((hardStop_pct/100.0)   * price) / mintick)) : na
exitLoss = not na(hardStopTicks) ? hardStopTicks : lossTicks

varip float breakevenPriceLong = na
varip float breakevenPriceShort = na

if noPosition
    breakevenPriceLong := na
    breakevenPriceShort := na

longEntryPrice = close + slippageTicks * mintick
shortEntryPrice = close - slippageTicks * mintick

// ───────────────────────────── EXECUTION ─────────────────────────────
// Entries
if finalLongConfirmed and validQty
    strategy.entry("LONG", strategy.long, qty = qty, limit = longEntryPrice)
    breakevenPriceLong := na

if finalShortConfirmed and validQty
    strategy.entry("SHORT", strategy.short, qty = qty, limit = shortEntryPrice)
    breakevenPriceShort := na

// Exits — BE triggers only on confirmed bars
if strategy.position_size > 0
    avgPrice = strategy.position_avg_price
    tp1Level = avgPrice + (slATRmult * atr * tp1Mult)
    if barstate.isconfirmed and moveToBE and high >= tp1Level and na(breakevenPriceLong)
        breakevenPriceLong := avgPrice

    if not na(trailingTicks)
        strategy.exit("LX_trail", from_entry="LONG", trail_points=trailingTicks, trail_offset=trailingTicks/2, stop=exitLoss)
    else if not na(breakevenPriceLong)
        strategy.exit("LX_BE", from_entry="LONG", stop=breakevenPriceLong)
    else
        strategy.exit("LX", from_entry="LONG", profit=profitTicks, loss=exitLoss)

if strategy.position_size < 0
    avgPrice = strategy.position_avg_price
    tp1Level = avgPrice - (slATRmult * atr * tp1Mult)
    if barstate.isconfirmed and moveToBE and low <= tp1Level and na(breakevenPriceShort)
        breakevenPriceShort := avgPrice

    if not na(trailingTicks)
        strategy.exit("SX_trail", from_entry="SHORT", trail_points=trailingTicks, trail_offset=trailingTicks/2, stop=exitLoss)
    else if not na(breakevenPriceShort)
        strategy.exit("SX_BE", from_entry="SHORT", stop=breakevenPriceShort)
    else
        strategy.exit("SX", from_entry="SHORT", profit=profitTicks, loss=exitLoss)

// ───────────────────────────── PLOTS & ALERTS ─────────────────────────
plot(fEMA, color=color.orange, title="Fast EMA", linewidth=1)
plot(sEMA, color=color.blue, title="Slow EMA", linewidth=1)
plot(useHTF ? htfEMA : na, color=color.teal, title="HTF EMA", linewidth=2)

plot(strategy.equity, title="Equity Curve", color=color.fuchsia, linewidth=2)
bgcolor(finalLongConfirmed ? color.new(color.green, 90) : finalShortConfirmed ? color.new(color.red, 90) : na)
plotshape(pivotH and barstate.isconfirmed, "Pivot High", shape.triangledown, location.abovebar, color=color.red, size=size.tiny)
plotshape(pivotL and barstate.isconfirmed, "Pivot Low", shape.triangleup, location.belowbar, color=color.green, size=size.tiny)
plotchar(finalLongConfirmed, "▲ LONG", "", location.belowbar, color=color.green, size=size.tiny)
plotchar(finalShortConfirmed, "▼ SHORT", "", location.abovebar, color=color.red, size=size.tiny)

// ALERTS — SAFE & UNIVERSAL
alertMessage(side) =>
    '{"ticker":"' + syminfo.ticker + '","side":"' + side + '","qty":' + str.tostring(qty) + ',"lev":' + str.tostring(leverage) + ',"risk_usd":' + str.tostring(dollarRisk) + '}'

if finalLongConfirmed
    alert(alertMessage("BUY"), freq = alert.freq_once_per_bar)
if finalShortConfirmed
    alert(alertMessage("SELL"), freq = alert.freq_once_per_bar)

// ─────────────── DEBUG: BAR CONFIRMATION & SIGNAL TIMING ──────────────
plotshape(barstate.isconfirmed, title="Bar Confirmed", location=location.belowbar, color=color.gray, style=shape.circle, size=size.tiny)
plotshape(finalLongConfirmed, title="LONG Signal (Confirmed)", location=location.belowbar, color=color.green, style=shape.labelup, text="LONG", size=size.small)
plotshape(finalShortConfirmed, title="SHORT Signal (Confirmed)", location=location.abovebar, color=color.red, style=shape.labeldown, text="SHORT", size=size.small)
