//@version=5
strategy('Enhanced Bollinger Bands Strategy v2.1 - Webhook Optimized & Fixed', overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=10, max_bars_back=500)

// Timing settings (historical)
FromMonth = input.int(defval=1, title='From Month', minval=1, maxval=12)
FromDay = input.int(defval=1, title='From Day', minval=1, maxval=31)
FromYear = input.int(defval=2020, title='From Year', minval=2020)
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12)
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31)
ToYear = input.int(defval=2999, title='To Year', minval=2020)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() => time[1] >= start and time[1] <= finish  // FIXED: [1] confirmed

// Define input parameters
length = input.int(20, minval=1, title='SMA Length')
mult = input.float(2.0, minval=0.001, maxval=50, title='Standard Deviation Multiplier')
trailingPercent = input.float(1.5, title='Trailing Stop (%)')
pyramidMax = input.int(2, title='Max Pyramid Orders')
percentPerOrder = input.float(10, title='Percent of Equity per Order')
useTrendFilter = input.bool(true, title='Use MACD Trend Filter')
useConfirmationDelay = input.bool(true, title='Use Confirmation Delay')
confirmationDelayBars = input.int(2, title='Confirmation Delay Bars', minval=0)
useDynamicTakeProfit = input.bool(true, title='Use Dynamic Take Profit')
atrMultiplier = input.float(1.5, title='ATR Multiplier for Take Profit')
atrLength = input.int(14, title='ATR Length')

// Cooldown (Global 2-bar cooldown between entries)
cooldownBars = input.int(2, title='Cooldown Bars Between Entries', minval=0)
var int lastEntryBar = na
cooldownPassed = na(lastEntryBar) or bar_index - lastEntryBar > cooldownBars

// MACD for trend filtering (unpack tuple first, then shift)
[macdLineRaw, signalLineRaw, _] = ta.macd(close[1], 12, 26, 9)  // FIXED: Unpack without shift
macdLine = macdLineRaw[1]  // FIXED: Shift each
signalLine = signalLineRaw[1]  // FIXED: Shift each
longTrend = macdLine > signalLine
shortTrend = macdLine < signalLine

// Bollinger Bands calculation (historical [1])
basisRaw = ta.sma(close[1], length)  // FIXED: [1]
devRaw = mult * ta.stdev(close[1], length)  // FIXED: [1]
basis = basisRaw[1]
dev = devRaw[1]
upper = basis + dev
lower = basis - dev

// ATR for dynamic take profit (historical [1])
atrValueRaw = ta.atr(atrLength)  // FIXED: [1]
atrValue = atrValueRaw[1]

// Bar counting for confirmation delays (historical)
var int longBars = 0
var int shortBars = 0

longCondition = ta.crossover(close[1], lower) and (useTrendFilter ? longTrend : true)  // FIXED: [1]
shortCondition = ta.crossunder(close[1], upper) and (useTrendFilter ? shortTrend : true)  // FIXED: [1]

longBars := longCondition ? 1 : longBars > 0 and longBars < confirmationDelayBars ? longBars + 1 : 0
shortBars := shortCondition ? 1 : shortBars > 0 and shortBars < confirmationDelayBars ? shortBars + 1 : 0

longConfirmed = longBars >= confirmationDelayBars
shortConfirmed = shortBars >= confirmationDelayBars

// Final entry conditions with confirmation (immediate on signal)
longEntryCondition = barstate.isconfirmed and longConfirmed and cooldownPassed and window()  // FIXED: Confirmed, immediate entry
shortEntryCondition = barstate.isconfirmed and shortConfirmed and cooldownPassed and window()  // FIXED: Confirmed, immediate entry

// Exit conditions (delayed to next bar or subsequent)
var bool pendingExitLong = false
var bool pendingExitShort = false

// Trigger pending exit on signal (but execute next bar)
if strategy.position_size > 0 and shortEntryCondition
    pendingExitLong := true  // Flag for next bar

if strategy.position_size < 0 and longEntryCondition
    pendingExitShort := true  // Flag for next bar

// Execute exits on next confirmed bar (real-time in subsequent bars)
longExitCondition = pendingExitLong and barstate.isconfirmed
shortExitCondition = pendingExitShort and barstate.isconfirmed

// Debugging outputs (confirmed)
if barstate.islast and bar_index % 100 == 0
    label.new(bar_index, high, text='Debug: LongCond ' + str.tostring(longCondition) + '\nShortCond ' + str.tostring(shortCondition) + '\nLongTrend ' + str.tostring(longTrend) + '\nShortTrend ' + str.tostring(shortTrend) + '\nLongConfirmed ' + str.tostring(longConfirmed) + '\nShortConfirmed ' + str.tostring(shortConfirmed) + '\nCooldown OK ' + str.tostring(cooldownPassed), color=color.new(color.navy, 90))

// Entry: Immediate on signal
if longEntryCondition
    strategy.entry('Long Entry', strategy.long)
    lastEntryBar := bar_index  // Update cooldown

if shortEntryCondition
    strategy.entry('Short Entry', strategy.short)
    lastEntryBar := bar_index  // Update cooldown

// Exit: Delayed to next bar
if longExitCondition
    strategy.close('Long Entry', comment="Delayed Exit Long")
    pendingExitLong := false

if shortExitCondition
    strategy.close('Short Entry', comment="Delayed Exit Short")
    pendingExitShort := false

// Trailing & Dynamic TP (on confirmed)
if strategy.position_size > 0
    strategy.exit('ExitLong', 'Long Entry', trail_price=close * (1 - trailingPercent / 100), trail_offset=close * (trailingPercent / 100))
    if useDynamicTakeProfit
        strategy.exit('TakeProfitLong', 'Long Entry', limit=close + atrValue * atrMultiplier)

if strategy.position_size < 0
    strategy.exit('ExitShort', 'Short Entry', trail_price=close * (1 + trailingPercent / 100), trail_offset=close * (trailingPercent / 100))
    if useDynamicTakeProfit
        strategy.exit('TakeProfitShort', 'Short Entry', limit=close - atrValue * atrMultiplier)

// Plot BB for visual
plot(basis, color=color.blue, title="Basis")
p1 = plot(upper, color=color.red, title="Upper")
p2 = plot(lower, color=color.green, title="Lower")
fill(p1, p2, color=color.new(color.purple, 95), title="BB Fill")

// Webhook Alerts (Separate for Entry/Exit - Copy to TV Alerts)
if longEntryCondition
    alert('{"symbol":"{{ticker}}","side":"buy","qty":"' + str.tostring(math.round(strategy.position_size * (percentPerOrder / 100), 3)) + '","price":"' + str.tostring(close) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

if shortEntryCondition
    alert('{"symbol":"{{ticker}}","side":"sell","qty":"' + str.tostring(math.round(math.abs(strategy.position_size) * (percentPerOrder / 100), 3)) + '","price":"' + str.tostring(close) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

// Exit Alerts (Reduce Only, Delayed)
if longExitCondition
    alert('{"symbol":"{{ticker}}","side":"sell","qty":"' + str.tostring(math.round(strategy.position_size, 3)) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)

if shortExitCondition
    alert('{"symbol":"{{ticker}}","side":"buy","qty":"' + str.tostring(math.round(math.abs(strategy.position_size), 3)) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_once_per_bar_close)
