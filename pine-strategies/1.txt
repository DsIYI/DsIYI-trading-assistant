//@version=6
strategy('Improved Pivot Point Strategy with EMA & Volume Filters, Reversal, Trailing Stop, and Consolidation Zones', overlay = true, default_qty_type = strategy.percent_of_equity, default_qty_value = 15, max_bars_back = 4999)

// Timing settings
FromMonth = input.int(defval = 4, title = 'From Month', minval = 1, maxval = 12)
FromDay = input.int(defval = 17, title = 'From Day', minval = 1, maxval = 31)
FromYear = input.int(defval = 2024, title = 'From Year', minval = 2024)
ToMonth = input.int(defval = 1, title = 'To Month', minval = 1, maxval = 12)
ToDay = input.int(defval = 1, title = 'To Day', minval = 1, maxval = 31)
ToYear = input.int(defval = 9999, title = 'To Year', minval = 2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish

// Pivot Point settings
pivotPeriod = input.int(defval = 5, title = 'Pivot Period', minval = 1, maxval = 50)
candleWait = input.int(defval = 1, title = 'Candle Wait Period', options = [1, 2, 3, 5])

// EMA and Volume Filter settings
emaLength = 9
volumeLookback = 50

// EMA and Volume calculation
emaValue = ta.ema(close, emaLength)
avgVolume = ta.sma(volume, volumeLookback)

// Pivot Point calculations
pivotHigh = ta.pivothigh(high, pivotPeriod, pivotPeriod)
pivotLow = ta.pivotlow(low, pivotPeriod, pivotPeriod)

// Plot Pivot Points
plotshape(series = pivotHigh, location = location.abovebar, color = color.red, style = shape.labeldown, title = 'Pivot High')
plotshape(series = pivotLow, location = location.belowbar, color = color.green, style = shape.labelup, title = 'Pivot Low')

// Check if Pivot Points have stayed for the specified candleWait period
var float lastPivotHigh = na
var float lastPivotLow = na
var int barsSincePivotHigh = na
var int barsSincePivotLow = na

if not na(pivotHigh)
    lastPivotHigh := high[pivotPeriod]
    barsSincePivotHigh := 0
    barsSincePivotHigh
else
    barsSincePivotHigh := na(barsSincePivotHigh) ? na : barsSincePivotHigh + 1
    barsSincePivotHigh

if not na(pivotLow)
    lastPivotLow := low[pivotPeriod]
    barsSincePivotLow := 0
    barsSincePivotLow
else
    barsSincePivotLow := na(barsSincePivotLow) ? na : barsSincePivotLow + 1
    barsSincePivotLow

pivotHighCondition = not na(barsSincePivotHigh) and barsSincePivotHigh == candleWait
pivotLowCondition = not na(barsSincePivotLow) and barsSincePivotLow == candleWait

// Consolidation Zones settings
prd = input.int(defval = 10, title = 'Loopback Period', minval = 2, maxval = 50)
conslen = input.int(defval = 5, title = 'Min Consolidation Length', minval = 2, maxval = 20)
paintcons = input(defval = true, title = 'Paint Consolidation Area')
zonecol = input.color(defval = color.new(color.blue, 70), title = 'Zone Color')

float hb_ = na
float lb_ = na
if ta.highestbars(high, prd) == 0
    hb_ := high
    hb_
if ta.lowestbars(low, prd) == 0
    lb_ := low
    lb_

var int dir = 0
float zz = na
float pp = na

dir := not na(hb_) ? 1 : not na(lb_) ? -1 : dir
if not na(hb_) and not na(lb_)
    if dir == 1
        zz := hb_
        zz
    else
        zz := lb_
        zz
else
    zz := na
    zz

var int conscnt = 0
var float condhigh = na
var float condlow = na
float H_ = ta.highest(conslen)
float L_ = ta.lowest(conslen)
var line upline = na
var line dnline = na
bool breakoutup = false
bool breakoutdown = false

if ta.change(pp) != 0
    if conscnt > conslen
        if pp > condhigh
            breakoutup := true
            breakoutup
        if pp < condlow
            breakoutdown := true
            breakoutdown
    if conscnt > 0 and pp <= condhigh and pp >= condlow
        conscnt := conscnt + 1
        conscnt
    else
        conscnt := 0
        conscnt
else
    conscnt := conscnt + 1
    conscnt

if conscnt >= conslen
    if conscnt == conslen
        condhigh := H_
        condlow := L_
        condlow
    else
        line.delete(upline)
        line.delete(dnline)
        condhigh := math.max(condhigh, high)
        condlow := math.min(condlow, low)
        condlow

    upline := line.new(x1 = bar_index - conscnt, y1 = condhigh, x2 = bar_index, y2 = condhigh, color = color.red, style = line.style_dashed)
    dnline := line.new(x1 = bar_index - conscnt, y1 = condlow, x2 = bar_index, y2 = condlow, color = color.lime, style = line.style_dashed)
    dnline

fill(plot(condhigh, color = na, style = plot.style_stepline), plot(condlow, color = na, style = plot.style_stepline), color = paintcons and conscnt > conslen ? zonecol : color.new(color.white, 100))

// Entry conditions based on Pivot Points, EMA, volume, and consolidation zones
longConditionPivot = pivotLowCondition and window() and not breakoutup and not breakoutdown
shortConditionPivot = pivotHighCondition and window() and not breakoutup and not breakoutdown

longConditionEMA = close > emaValue and volume > avgVolume and window() and not breakoutup and not breakoutdown
shortConditionEMA = close < emaValue and volume > avgVolume and window() and not breakoutup and not breakoutdown

// Additional filter to avoid trading in choppy markets (consolidation)
consolidationZoneActive = conscnt >= conslen

// Slippage and Spread consideration
slippage = input.float(defval = 0.1, title = 'Slippage (%)')
spread = input.float(defval = 0.1, title = 'Spread (%)')

// Stop-Loss and Trailing Stop settings
stopLossPercent = input.float(1.0, 'Stop-Loss (%)', minval = 0.1, maxval = 10.0) + slippage + spread
trailingStop = input.float(1.5, 'Trailing Stop (%)', step = 0.1) + slippage + spread

// Calculate stop-loss levels
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// Execute trades with reversible signals and regular stop-loss
if (longConditionPivot or longConditionEMA) and not consolidationZoneActive
    strategy.close('Short Entry', comment = 'Close Short on Long Signal')
    if longConditionPivot
        strategy.entry('Long Entry Pivot', strategy.long)
        strategy.exit('Close Long Pivot', 'Long Entry Pivot', stop = longStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)
    if longConditionEMA
        strategy.entry('Long Entry EMA', strategy.long)
        strategy.exit('Close Long EMA', 'Long Entry EMA', stop = longStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

if (shortConditionPivot or shortConditionEMA) and not consolidationZoneActive
    strategy.close('Long Entry', comment = 'Close Long on Short Signal')
    if shortConditionPivot
        strategy.entry('Short Entry Pivot', strategy.short)
        strategy.exit('Close Short Pivot', 'Short Entry Pivot', stop = shortStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)
    if shortConditionEMA
        strategy.entry('Short Entry EMA', strategy.short)
        strategy.exit('Close Short EMA', 'Short Entry EMA', stop = shortStopLoss, trail_points = trailingStop / 100 * close, trail_offset = trailingStop / 100 * close)

// Debugging: Plot conditions
plotshape(series = longConditionPivot, title = 'Long Condition Pivot', location = location.belowbar, color = color.blue, style = shape.labelup, size = size.small)
plotshape(series = shortConditionPivot, title = 'Short Condition Pivot', location = location.abovebar, color = color.orange, style = shape.labeldown, size = size.small)

plotshape(series = longConditionEMA, title = 'Long Condition EMA', location = location.belowbar, color = color.green, style = shape.labelup, size = size.small)
plotshape(series = shortConditionEMA, title = 'Short Condition EMA', location = location.abovebar, color = color.red, style = shape.labeldown, size = size.small)
