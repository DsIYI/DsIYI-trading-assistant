//@version=5
strategy("Custom Pivot Points and Timeframe Strategy", overlay=true, max_bars_back=500)

// Inputs
length = input.int(50, 'Pivot Length', minval=1, maxval=100, step=1)
trailStopPercent = input.float(1.5, 'Trailing Stop (%)', minval=0.1, maxval=10.0, step=0.1)
stopLossPercent = input.float(1.0, 'Stop Loss (%)', minval=0.1, maxval=10.0, step=0.1)

// Timing settings
FromMonth = input.int(4, 'From Month', minval=1, maxval=12)
FromDay = input.int(17, 'From Day', minval=1, maxval=31)
FromYear = input.int(2024, 'From Year', minval=2024)
ToMonth = input.int(1, 'To Month', minval=1, maxval=12)
ToDay = input.int(1, 'To Day', minval=1, maxval=31)
ToYear = input.int(9999, 'To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish

// Color Change Detection Function
colorChange() =>
    change = 0
    if close[1] < open[1] and close > open
        change := 1
    else if close[1] > open[1] and close < open
        change := -1
    change

// Detect pivot points (confirmed)
ph = ta.pivothigh(length, length)
pl = ta.pivotlow(length, length)

// Entry conditions based on pivot confirmation and bar color change
buyCondition = not na(pl) and colorChange() == 1 and window()
sellCondition = not na(ph) and colorChange() == -1 and window()

// Strategy Execution Logic
if (buyCondition)
    strategy.close_all(comment="Close All on Buy Signal")  // Close all positions before a new buy
    strategy.entry("Buy Entry", strategy.long)
    strategy.exit("Exit Buy", "Buy Entry", stop=close * (1 - stopLossPercent / 100), trail_price=close * (1 - trailStopPercent / 100), trail_offset=0)

if (sellCondition)
    strategy.close_all(comment="Close All on Sell Signal")  // Close all positions before a new sell
    strategy.entry("Sell Entry", strategy.short)
    strategy.exit("Exit Sell", "Sell Entry", stop=close * (1 + stopLossPercent / 100), trail_price=close * (1 + trailStopPercent / 100), trail_offset=0)

// Plot pivot points and confirmations
plotshape(series=pl and buyCondition, location=location.belowbar, color=color.green, style=shape.triangleup, title="Buy Signal")
plotshape(series=ph and sellCondition, location=location.abovebar, color=color.red, style=shape.triangledown, title="Sell Signal")
