//@version=5
strategy('Refined Supertrend Strategy with EMA & Volume Filters and Trailing Stop - Fixed', overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15)

// --- Fixed Start/End Date (live-friendly) ---
FromMonth = input.int(defval=4, title='From Month', minval=1, maxval=12)
FromDay   = input.int(defval=17, title='From Day', minval=1, maxval=31)
FromYear  = input.int(defval=2024, title='From Year', minval=2024)
ToMonth   = input.int(defval=12, title='To Month', minval=1, maxval=12)  // FIXED: Current for live
ToDay     = input.int(defval=31, title='To Day', minval=1, maxval=31)
ToYear    = input.int(defval=9999, title='To Year', minval=2024)

start  = timestamp(FromYear, FromMonth, FromDay, 0, 0)
finish = timestamp(ToYear,   ToMonth,   ToDay,   23, 59)

window() => time >= start and time <= finish

// --- Rolling N-Day Lookback ---
daysBack = input.int(30, 'Days to Backtest', minval=1)
oneDay   = 24 * 60 * 60 * 1000
lookback = daysBack * oneDay
startTime       = timenow - lookback
inLookbackWin   = time >= startTime

// --- Supertrend inputs ---
atrPeriod     = input(7, 'ATR Length')
factor        = input.float(0.5, 'Factor', step=0.01)
trailingStop  = input.float(1.5, 'Trailing Stop (%)', step=0.1)

// NEW: Anti-Whipsaw (prevents rapid open/exit)
signalCooldown = input.int(3, title="Cooldown Bars Between Signals", minval=1)
var int lastSignalBar = na

// Supertrend calculation
[supertrend, direction] = ta.supertrend(factor, atrPeriod)

// Example logic (with filters)
longCondition  = ta.change(direction) < 0
shortCondition = ta.change(direction) > 0
longVolume     = volume > ta.sma(volume, 20)
shortVolume    = longVolume // Just placeholder

// Cooldown & confirmed
cooldownPassed = na(lastSignalBar) or bar_index - lastSignalBar > signalCooldown

// Incorporate BOTH time filters in final signals (with guards)
finalLongCondition  = longCondition and longVolume and window() and inLookbackWin and cooldownPassed and barstate.isconfirmed  // FIXED: Confirmed + cooldown
finalShortCondition = shortCondition and shortVolume and window() and inLookbackWin and cooldownPassed and barstate.isconfirmed  // FIXED: Confirmed + cooldown

// Track entry for fixed trail
var float longEntry = na
var float shortEntry = na

// Trade logic (gated closes)
if finalLongCondition
    if strategy.position_size < 0  // FIXED: Only close if short
        strategy.close('Short Entry', comment="Close Short on Long")
    strategy.entry('Long Entry', strategy.long)
    longEntry := close  // FIXED: Hold for trail
    lastSignalBar := bar_index

if finalShortCondition
    if strategy.position_size > 0  // FIXED: Only close if long
        strategy.close('Long Entry', comment="Close Long on Short")
    strategy.entry('Short Entry', strategy.short)
    shortEntry := close  // FIXED: Hold for trail
    lastSignalBar := bar_index

// FIXED: Proper Trailing Exits (fixed % from entry, no dynamic close)
if strategy.position_size > 0
    trailStopLong = longEntry * (1 - trailingStop / 100)  // FIXED: From entry
    strategy.exit('Close Long', 'Long Entry', trail_price=trailStopLong, trail_offset=trailingStop / 100 * syminfo.mintick * syminfo.pointvalue)  // FIXED: Proper units

if strategy.position_size < 0
    trailStopShort = shortEntry * (1 + trailingStop / 100)  // FIXED: From entry
    strategy.exit('Close Short', 'Short Entry', trail_price=trailStopShort, trail_offset=trailingStop / 100 * syminfo.mintick * syminfo.pointvalue)  // FIXED: Proper units

// Debugging: Plot conditions
plotshape(series=finalLongCondition, title="Long Signal", location=location.belowbar, color=color.green, style=shape.labelup, size=size.small)
plotshape(series=finalShortCondition, title="Short Signal", location=location.abovebar, color=color.red, style=shape.labeldown, size=size.small)
