//@version=6
strategy('Enhanced Debugging Strategy', overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=25)

// Timing settings
FromMonth = input.int(defval=1, title='From Month', minval=1, maxval=12, group='Timing')
FromDay = input.int(defval=1, title='From Day', minval=1, maxval=31, group='Timing')
FromYear = input.int(defval=2024, title='From Year', minval=2024, group='Timing')
ToMonth = input.int(defval=12, title='To Month', minval=1, maxval=12, group='Timing')
ToDay = input.int(defval=31, title='To Day', minval=1, maxval=31, group='Timing')
ToYear = input.int(defval=2024, title='To Year', minval=2024, group='Timing')

start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)

window() =>
    time >= start and time <= finish

// Indicator settings
emaLength = input.int(defval=6, title='EMA Length', minval=1, maxval=100)
rsiLength = input.int(defval=14, title='RSI Length', minval=1, maxval=50)
macdFastLength = input.int(defval=12, title='MACD Fast Length', minval=1, maxval=50)
macdSlowLength = input.int(defval=26, title='MACD Slow Length', minval=1, maxval=50)
macdSignalSmoothing = input.int(defval=9, title='MACD Signal Smoothing', minval=1, maxval=50)

// Risk management settings
stopLossPercent = input.float(defval=1.8, title='Stop Loss (%)', minval=0.1, maxval=10.0, step=0.1, group='Risk Management')
trailingStopPercent = input.float(defval=0.8, title='Trailing Stop (%)', step=0.1, group='Risk Management')

// Calculations
emaValue = ta.ema(close, emaLength)
rsiValue = ta.rsi(close, rsiLength)
[macdLine, signalLine, _] = ta.macd(close, macdFastLength, macdSlowLength, macdSignalSmoothing)

// Simplified Signal
buySignal = close > emaValue and rsiValue < 70 and macdLine > signalLine
sellSignal = close < emaValue and rsiValue > 30 and macdLine < signalLine

// Debugging Plots
plot(emaValue, title='EMA Value', color=color.blue)
plot(rsiValue, title='RSI Value', color=color.purple)
plot(macdLine, title='MACD Line', color=color.green)
plot(signalLine, title='Signal Line', color=color.red)

plotshape(buySignal ? close : na, style=shape.triangleup, color=color.green, location=location.belowbar, title="Buy Signal")
plotshape(sellSignal ? close : na, style=shape.triangledown, color=color.red, location=location.abovebar, title="Sell Signal")

// Stop-loss and trailing stop calculations
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)
trailingStop = trailingStopPercent / 100 * close

// Trading logic
if buySignal and window()
    strategy.close('Short', comment='Close Short on Buy Signal')
    strategy.entry('Long', strategy.long)
    strategy.exit('Close Long', 'Long', stop=longStopLoss, trail_points=trailingStop)

if sellSignal and window()
    strategy.close('Long', comment='Close Long on Sell Signal')
    strategy.entry('Short', strategy.short)
    strategy.exit('Close Short', 'Short', stop=shortStopLoss, trail_points=trailingStop)
