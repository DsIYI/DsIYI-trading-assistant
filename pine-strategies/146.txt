//@version=6
strategy(title           = 'Enhanced Pivot Point Strategy with EMA, Volume, RSI & MACD Filters, Non-Reversal + Manual Side Switch',overlay = true,default_qty_type= strategy.percent_of_equity,default_qty_value=15,max_bars_back   = 4999)

//  ──────────────────────────────
//  1. INPUTS & SETTINGS
//  ──────────────────────────────

// Group: Timing
FromMonth   = input.int(defval=7,  title='From Month', minval=1, maxval=12, group='Timing')
FromDay     = input.int(defval=5,  title='From Day',   minval=1, maxval=31, group='Timing')
FromYear    = input.int(defval=2024, title='From Year', minval=2024,        group='Timing')
ToMonth     = input.int(defval=1,  title='To Month',   minval=1, maxval=12, group='Timing')
ToDay       = input.int(defval=1,  title='To Day',     minval=1, maxval=31, group='Timing')
ToYear      = input.int(defval=9999, title='To Year',  minval=2024,         group='Timing')

start       = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish      = timestamp(ToYear,   ToMonth,   ToDay,   23, 59)

// Helper function for date window
window() =>
    time >= start and time <= finish

// Group: Pivot Points
pivotPeriod = input.int(defval=5,  title='Pivot Period',      minval=1,  maxval=50, group='Pivot Points')
candleWait  = input.int(defval=1,  title='Candle Wait Period', options=[1,2,3,5],   group='Pivot Points')

// Group: EMA & Volume Filter
emaLength      = input.int(defval=9,   title='EMA Length',           minval=1, maxval=100, group='EMA & Volume Filter')
volumeLookback = input.int(defval=50,  title='Volume Lookback Period',minval=1, maxval=100, group='EMA & Volume Filter')

// Group: RSI Filter
rsiLength = input.int(defval=14, title='RSI Length', minval=1, maxval=50, group='RSI Filter')

// Group: MACD Filter
macdFastLength       = input.int(defval=12, title='MACD Fast Length',      minval=1,  maxval=50, group='MACD Filter')
macdSlowLength       = input.int(defval=26, title='MACD Slow Length',      minval=1,  maxval=50, group='MACD Filter')
macdSignalSmoothing  = input.int(defval=9,  title='MACD Signal Smoothing', minval=1,  maxval=50, group='MACD Filter')

// Group: Stop-Loss & Trailing Stop
stopLossPercent = input.float(defval=1.0, title='Stop-Loss (%)', minval=0.1, maxval=10.0, step=0.1, group='Stop-Loss & Trailing Stop')
trailingStop    = input.float(defval=1.5, title='Trailing Stop (%)',          step=0.1,              group='Stop-Loss & Trailing Stop')

// Group: Trade Direction Control
tradeDirection = input.string(defval   = 'Both',title    = 'Trade Direction Allowed',options  = ['Buy', 'Sell', 'Both'],group    = 'Trade Mode')

//  ──────────────────────────────
//  2. CALCULATIONS: Pivots, EMA, Volume, RSI, MACD
//  ──────────────────────────────

// EMA and Volume
emaValue   = ta.ema(close, emaLength)
avgVolume  = ta.sma(volume, volumeLookback)

// Pivot Points
pivotHigh  = ta.pivothigh(high, pivotPeriod, pivotPeriod)
pivotLow   = ta.pivotlow(low,   pivotPeriod, pivotPeriod)

// Plot Pivot Points on chart
plotshape(series   = pivotHigh,location = location.abovebar,color = color.red,style = shape.labeldown,title = 'Pivot High')
plotshape(series   = pivotLow,location = location.belowbar,color = color.green,style = shape.labelup,title = 'Pivot Low')

// Track pivot appearance for candleWait
var float lastPivotHigh        = na
var float lastPivotLow         = na
var int   barsSincePivotHigh   = na
var int   barsSincePivotLow    = na

if not na(pivotHigh)
    lastPivotHigh      := high[pivotPeriod]
    barsSincePivotHigh := 0
else
    barsSincePivotHigh := na(barsSincePivotHigh) ? na : barsSincePivotHigh + 1

if not na(pivotLow)
    lastPivotLow      := low[pivotPeriod]
    barsSincePivotLow := 0
else
    barsSincePivotLow := na(barsSincePivotLow) ? na : barsSincePivotLow + 1

pivotHighCondition = not na(barsSincePivotHigh) and barsSincePivotHigh == candleWait
pivotLowCondition  = not na(barsSincePivotLow)  and barsSincePivotLow  == candleWait

// RSI
rsiValue = ta.rsi(close, rsiLength)

// MACD
[macdLine, signalLine, _] = ta.macd(close, macdFastLength, macdSlowLength, macdSignalSmoothing)

//  ──────────────────────────────
//  3. ENTRY CONDITIONS
//  ──────────────────────────────

// Combine your filters any way you like. 
// Each group below uses OR across filters, but you can refine further as needed.
longConditionPivot = pivotLowCondition  and window()
shortConditionPivot= pivotHighCondition and window()

longConditionEMA   = close > emaValue   and (volume > avgVolume) and window()
shortConditionEMA  = close < emaValue   and (volume > avgVolume) and window()

longConditionRSI   = rsiValue < 30      and window()
shortConditionRSI  = rsiValue > 70      and window()

longConditionMACD  = macdLine > signalLine  and window()
shortConditionMACD = macdLine < signalLine  and window()

//  ──────────────────────────────
//  4. STOP-LOSS & TRAILING STOP
//  ──────────────────────────────

// Calculate stop-loss levels
longStopLoss  = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

//  ──────────────────────────────
//  5. NON‑REVERSAL TRADE EXECUTION
//     (Close existing position first, then open new)
//  ──────────────────────────────

// --- LONG SIDE: If conditions for a long are met & user allows Buy or Both
if (longConditionPivot or longConditionEMA or longConditionRSI or longConditionMACD) and (tradeDirection == 'Buy' or tradeDirection == 'Both')
    // 1) Close all short entries before opening a new long
    strategy.close('Short Entry Pivot', comment='Close Short Pivot → Long Signal')
    strategy.close('Short Entry EMA',   comment='Close Short EMA → Long Signal')
    strategy.close('Short Entry RSI',   comment='Close Short RSI → Long Signal')
    strategy.close('Short Entry MACD',  comment='Close Short MACD → Long Signal')
    
    // 2) Now open new long entries
    if longConditionPivot
        strategy.entry('Long Entry Pivot', strategy.long)
        strategy.exit('Close Long Pivot','Long Entry Pivot',stop = longStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if longConditionEMA
        strategy.entry('Long Entry EMA', strategy.long)
        strategy.exit('Close Long EMA','Long Entry EMA',stop = longStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if longConditionRSI
        strategy.entry('Long Entry RSI', strategy.long)
        strategy.exit('Close Long RSI','Long Entry RSI',stop = longStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if longConditionMACD
        strategy.entry('Long Entry MACD', strategy.long)
        strategy.exit('Close Long MACD','Long Entry MACD',stop = longStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)

// --- SHORT SIDE: If conditions for a short are met & user allows Sell or Both
if (shortConditionPivot or shortConditionEMA or shortConditionRSI or shortConditionMACD) and (tradeDirection == 'Sell' or tradeDirection == 'Both')
    // 1) Close all long entries before opening a new short
    strategy.close('Long Entry Pivot', comment='Close Long Pivot → Short Signal')
    strategy.close('Long Entry EMA',   comment='Close Long EMA → Short Signal')
    strategy.close('Long Entry RSI',   comment='Close Long RSI → Short Signal')
    strategy.close('Long Entry MACD',  comment='Close Long MACD → Short Signal')
    
    // 2) Now open new short entries
    if shortConditionPivot
        strategy.entry('Short Entry Pivot', strategy.short)
        strategy.exit('Close Short Pivot','Short Entry Pivot',stop = shortStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if shortConditionEMA
        strategy.entry('Short Entry EMA', strategy.short)
        strategy.exit('Close Short EMA','Short Entry EMA',stop = shortStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if shortConditionRSI
        strategy.entry('Short Entry RSI', strategy.short)
        strategy.exit('Close Short RSI','Short Entry RSI',stop = shortStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)
    if shortConditionMACD
        strategy.entry('Short Entry MACD', strategy.short)
        strategy.exit('Close Short MACD','Short Entry MACD',stop = shortStopLoss,trail_points = trailingStop / 100 * close,trail_offset = trailingStop / 100 * close)

//  ──────────────────────────────
//  6. DEBUG PLOTS
//  ──────────────────────────────
plotshape(series   = longConditionPivot,title  = 'Long Condition Pivot',location = location.belowbar,color = color.new(color.blue, 0),style = shape.labelup,size = size.small)
plotshape(series   = shortConditionPivot,title = 'Short Condition Pivot',location = location.abovebar,color = color.new(color.orange, 0),style = shape.labeldown,size = size.small)

plotshape(series   = longConditionEMA,title   = 'Long Condition EMA',location = location.belowbar,color  = color.new(color.green, 0),style  = shape.labelup,size  = size.small)
plotshape(series   = shortConditionEMA,title  = 'Short Condition EMA',location = location.abovebar,color = color.new(color.red, 0),style  = shape.labeldown,size  = size.small)

plotshape(series   = longConditionRSI,title    = 'Long Condition RSI',location = location.belowbar,color = color.new(color.purple, 0),style = shape.labelup,size  = size.small)
plotshape(series   = shortConditionRSI,title   = 'Short Condition RSI',location = location.abovebar,color= color.new(color.yellow, 0),style = shape.labeldown,size = size.small)

plotshape(series   = longConditionMACD,title   = 'Long Condition MACD',location = location.belowbar,color= color.new(color.blue, 0),style   = shape.labelup,size   = size.small)
plotshape(series   = shortConditionMACD,title  = 'Short Condition MACD',location = location.abovebar,color= color.new(color.orange, 0),style = shape.labeldown,size = size.small)
