//@version=5
strategy("Supertrend Strategy with EMA & Volume Filters v1.6 - Immediate SL & Delayed Other Exits", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=15, max_bars_back=4999)

// Timing settings
FromMonth = input.int(defval=4, title='From Month', minval=1, maxval=12)
FromDay = input.int(defval=17, title='From Day', minval=1, maxval=31)
FromYear = input.int(defval=2024, title='From Year', minval=2024)
ToMonth = input.int(defval=1, title='To Month', minval=1, maxval=12)
ToDay = input.int(defval=1, title='To Day', minval=1, maxval=31)
ToYear = input.int(defval=9999, title='To Year', minval=2024)
start = timestamp(FromYear, FromMonth, FromDay, 00, 00)
finish = timestamp(ToYear, ToMonth, ToDay, 23, 59)
window() =>
    time >= start and time <= finish ? true : false

// Supertrend inputs
atrPeriod = input(2, "ATR Length")
factor = input.float(0.09, "Factor", step=0.01)
trailingStop = input.float(1.5, "Trailing Stop (%)", step=0.1)
stopLossPercent = input.float(1.0, "Stop-Loss (%)", minval=0.1, maxval=10.0)

// EMA and Volume Filter settings
emaLength = 9
volumeLookback = 50

// EMA and Volume calculation
emaValue = ta.ema(close, emaLength)
avgVolume = ta.sma(volume, volumeLookback)

// Supertrend calculation
[supertrend, direction] = ta.supertrend(factor, atrPeriod)

// Entry conditions based on Supertrend, EMA, and volume
longCondition = (ta.change(direction) < 0) and (close > emaValue) and (volume > avgVolume)
shortCondition = (ta.change(direction) > 0) and (close < emaValue) and (volume > avgVolume)

// Combine conditions for final entry signals
finalLongCondition = longCondition and window()
finalShortCondition = shortCondition and window()

// Calculate stop-loss levels
longStopLoss = close * (1 - stopLossPercent / 100)
shortStopLoss = close * (1 + stopLossPercent / 100)

// IMPROVED: Dynamic Symbol (removes exchange prefix, e.g., "BYBIT:" -> "TAOUSDT.P")
alertSymbol = str.replace(syminfo.ticker, syminfo.prefix + ":", "")

// Qty Calculation for Alerts (Fixed: Based on Equity % / Price)
qtyPercent = 0.15 // 15% of equity (matches default_qty_value)
var float alertQty = na
if finalLongCondition or finalShortCondition
    alertQty := math.round((strategy.equity * qtyPercent) / close, 3) // Qty = (Equity * %) / Price

// IMPROVED: Manual Trailing Stop Tracking for More Accurate Alerts
var float longTrailLevel = na
var float shortTrailLevel = na
var float longStopLossLevel = na  // NEW: Track SL level from entry
var float shortStopLossLevel = na // NEW: Track SL level from entry
var bool inLongPosition = false
var bool inShortPosition = false

// Update levels on entry
if finalLongCondition
    longTrailLevel := close * (1 - trailingStop / 100)
    longStopLossLevel := close * (1 - stopLossPercent / 100)  // Set SL level
    inLongPosition := true
    inShortPosition := false

if finalShortCondition
    shortTrailLevel := close * (1 + trailingStop / 100)
    shortStopLossLevel := close * (1 + stopLossPercent / 100) // Set SL level
    inShortPosition := true
    inLongPosition := false

// Check for trailing stop hits (manual detection for alerts - delayed)
longTrailHit = inLongPosition and close <= longTrailLevel
shortTrailHit = inShortPosition and close >= shortTrailLevel

// NEW: Manual Stop-Loss Hit Detection (immediate alerts)
longStopHit = inLongPosition and low <= longStopLossLevel
shortStopHit = inShortPosition and high >= shortStopLossLevel

// Track position changes for other exit alerts (delayed: opposite, trail)
var float lastPositionSize = na
lastPositionSize := nz(lastPositionSize, 0) // Initialize to 0 if na

// UPDATED: Delayed Exit Tracking (1 bar delay for NON-SL exits)
var bool pendingLongExit = false
var bool pendingShortExit = false
var float pendingCloseQty = na
var string pendingPnl = "0.00"

// Set pending flags for NON-SL exits (position change or trail - but not SL hit)
positionChanged = strategy.position_size != lastPositionSize
nonSlExitTriggered = positionChanged and lastPositionSize != 0 and not longStopHit and not shortStopHit
trailTriggeredLong = longTrailHit and inLongPosition and not longStopHit
trailTriggeredShort = shortTrailHit and inShortPosition and not shortStopHit

if nonSlExitTriggered or trailTriggeredLong or trailTriggeredShort
    pendingCloseQty := math.abs(lastPositionSize)
    pendingPnl := strategy.closedtrades > 0 ? str.tostring(strategy.closedtrades.profit(strategy.closedtrades - 1), "#.##") : "0.00"
    
    if lastPositionSize > 0 or trailTriggeredLong  // Pending LONG close (non-SL)
        pendingLongExit := true
        pendingShortExit := false
    else if lastPositionSize < 0 or trailTriggeredShort  // Pending SHORT close (non-SL)
        pendingShortExit := true
        pendingLongExit := false

// Execute trades with reversible signals and regular stop-loss
if finalLongCondition
    strategy.close("Short Entry", comment="Close Short on Long Signal") // Close existing short positions
    strategy.entry("Long Entry", strategy.long)
    strategy.exit("Close Long", "Long Entry", stop=longStopLossLevel, trail_points=trailingStop / 100 * close, trail_offset=trailingStop / 100 * close)

if finalShortCondition
    strategy.close("Long Entry", comment="Close Long on Short Signal") // Close existing long positions
    strategy.entry("Short Entry", strategy.short)
    strategy.exit("Close Short", "Short Entry", stop=shortStopLossLevel, trail_points=trailingStop / 100 * close, trail_offset=trailingStop / 100 * close)

// Reset position flags on exit (from strategy)
if strategy.position_size == 0
    inLongPosition := false
    inShortPosition := false

// UPDATED: Entry Alerts (Immediate - freq_all)
if finalLongCondition
    alert('{"symbol":"' + alertSymbol + '","side":"buy","qty":"' + str.tostring(alertQty) + '","price":"' + str.tostring(close) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_all)

if finalShortCondition
    alert('{"symbol":"' + alertSymbol + '","side":"sell","qty":"' + str.tostring(alertQty) + '","price":"' + str.tostring(close) + '","reduce_only":false,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity"}', alert.freq_all)

// NEW: Immediate Stop-Loss Exit Alerts (freq_all)
if longStopHit
    closeQty = math.abs(strategy.position_size)
    pnlValue = strategy.closedtrades > 0 ? str.tostring(strategy.closedtrades.profit(strategy.closedtrades - 1), "#.##") : "0.00"
    alert('{"symbol":"' + alertSymbol + '","side":"sell","qty":"' + str.tostring(closeQty) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity","pnl":"' + pnlValue + '","exit_type":"stop_loss"}', alert.freq_all)

if shortStopHit
    closeQty = math.abs(strategy.position_size)
    pnlValue = strategy.closedtrades > 0 ? str.tostring(strategy.closedtrades.profit(strategy.closedtrades - 1), "#.##") : "0.00"
    alert('{"symbol":"' + alertSymbol + '","side":"buy","qty":"' + str.tostring(closeQty) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity","pnl":"' + pnlValue + '","exit_type":"stop_loss"}', alert.freq_all)

// UPDATED: Delayed Exit Alerts (Send after 1 bar for NON-SL exits - freq_all for immediate send on next bar)
// Check pending flags and send alert, then reset
if pendingLongExit
    alert('{"symbol":"' + alertSymbol + '","side":"sell","qty":"' + str.tostring(pendingCloseQty) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity","pnl":"' + pendingPnl + '","exit_type":"normal"}', alert.freq_all)
    pendingLongExit := false
    pendingCloseQty := na

if pendingShortExit
    alert('{"symbol":"' + alertSymbol + '","side":"buy","qty":"' + str.tostring(pendingCloseQty) + '","price":"' + str.tostring(close) + '","reduce_only":true,"strategy_id":"your-strategy-id","amount_type":"absolute.quantity","pnl":"' + pendingPnl + '","exit_type":"normal"}', alert.freq_all)
    pendingShortExit := false
    pendingCloseQty := na

lastPositionSize := strategy.position_size // Update tracker
